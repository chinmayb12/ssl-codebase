270 void SoccerView::UpdateRobots ( SSL_DetectionFrame &detection )
271 {
272  int robots_blue_n = detection.robots_blue_size();
273  int robots_yellow_n = detection.robots_yellow_size();
274  int i,j,yellowj=0,bluej=0;
275  int team=teamBlue;
276  SSL_DetectionRobot robot;
277  for ( i = 0; i < robots_blue_n+robots_yellow_n; i++ )
278  {
279  if ( i<robots_blue_n )
280  {
281  robot = detection.robots_blue ( i );
282  team = teamBlue;
283  j=bluej;
284  }
285  else
286  {
287  robot = detection.robots_yellow ( i-robots_blue_n );
288  team = teamYellow;
289  j=yellowj;
290  }
291 
292  double x,y,orientation,conf =robot.confidence();
293  int id=NA;
294  if ( robot.has_robot_id() )
295  id = robot.robot_id();
296  else
297  id = NA;
298  x = robot.x();
299  y = robot.y();
300  if ( robot.has_orientation() )
301  orientation = robot.orientation() *180.0/M_PI;
302  else
303  orientation = NAOrientation;
304 
305  //seek to the next robot of the same camera and team colour
306  while ( j<robots.size() && ( robots[j]->key!=detection.camera_id() || robots[j]->teamID!=team ) )
307  j++;
308 
309  if ( j+1>robots.size() )
310  AddRobot ( new Robot ( x,y,orientation,team,id,detection.camera_id(),conf ) );
311 
312  robots[j]->SetPose ( x,y,orientation,conf );
313  QString label;
314 
315  if ( id!=NA )
316  label.setNum ( id,16 );
317  else
318  label = "?";
319  label = label.toUpper();
320  if ( label!=robots[j]->robotLabel )
321  robots[j]->robotLabel = label;
322 
323  j++;
324 
325  if ( i<robots_blue_n )
326  bluej=j;
327  else
328  yellowj=j;
329  }
330  for ( j=bluej;j<robots.size();j++ )
331  {
332  if ( robots[j]->key==detection.camera_id() && robots[j]->teamID==teamBlue )
333  robots[j]->conf=0.0;
334  }
335  for ( j=yellowj;j<robots.size();j++ )
336  {
337  if ( robots[j]->key==detection.camera_id() && robots[j]->teamID==teamYellow )
338  robots[j]->conf=0.0;
339  }
340  return;
341 }
