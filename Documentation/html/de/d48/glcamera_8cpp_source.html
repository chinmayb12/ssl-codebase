103 void GLCamera::lookAt(const vector3d & point) {
104  //1st compute yaw:
105  vector3d diff=(pos-point);
106  vector3d diff_flat(diff.x,0.0,diff.z);
107  double yaw=atan2(diff.x,diff.z);
108 
109  double pitch=shortest_angle(diff,diff_flat);
110 
111  if (diff.y > 0.0) pitch*=(-1.0);
112 
113  rot.setEuler(pitch,yaw,0.0);
114 }

172 void GLCamera::mouseAction ( QMouseEvent * event ) {
173  if ((event->buttons() & Qt::RightButton) != 0) {
174  QPoint offset=mouseStart_pan-event->pos();
175  if (pan_mode==GLCAM_PAN_POSITION_CENTRIC) {
176  forward.x=(pan_start_x+(double)offset.x()*max(getDistance(),1.0)/700.0);
177  forward.y=(pan_start_y-(double)offset.y()*max(getDistance(),1.0)/700.0);
178  } else if (pan_mode==GLCAM_PAN_FREE_FLY) {
179  vector3d trans;
180  trans.x=((double)offset.x()/10.0);
181  trans.y=(-(double)offset.y()/10.0);
182  trans.z=0.0;
183  trans=rot.rotateVectorByQuaternion(trans);
184  pos=pan_start_pos+trans;
185  }
186  }
187  if ((event->buttons() & Qt::LeftButton) != 0) {
188  QPoint offset=mouseStart_rotate-event->pos();
189  setEuler(p_start+(double)offset.y()/100.0,y_start+(double)offset.x()/100.0,r_start);
190  }
191 }
