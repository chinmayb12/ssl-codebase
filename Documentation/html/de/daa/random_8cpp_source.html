120 void Random::seed(uint32_t s)
121 {
122  int j;
123 
124  state[0]= s;
125  for(j=1; j<N; j++){
126  state[j] = (1812433253UL * (state[j-1] ^ (state[j-1] >> 30)) + j);
127  /* See Knuth TAOCP Vol2. 3rd Ed. P.106 for multiplier. */
128  /* In the previous versions, MSBs of the seed affect */
129  /* only MSBs of the array state[]. */

133  left = 1;

165 void Random::next_state()
166 {
167  uint32_t *p=state;
168  int j;
169 
170  /* if seed() or randomize() has not been called, */
171  /* a default initial seed is used */
172  if(left < 0) seed(5489UL);
173 
174  left = N;
175  next = state;
176 
177  for(j=N-M+1; --j; p++){
178  *p = p[M] ^ TWIST(p[0], p[1]);
179  }
180 
181  for(j=M; --j; p++){
182  *p = p[M-N] ^ TWIST(p[0], p[1]);
183  }
184
