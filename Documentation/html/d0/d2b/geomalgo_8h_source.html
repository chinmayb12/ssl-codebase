114 template <class num>
115 bool CalcCircle(vector2d<num> &cen,num &rad,
116  const vector2d<num> p1,
117  const vector2d<num> p2,
118  const vector2d<num> p3)
119 {
120  vector2d<num> a,b;
121  num da,db;
122 
123  // edge vectors
124  a = (p2 - p1);
125  b = (p3 - p2);
126 
127  // dot products of midpoint of each edge vector
128  da = a.dot(p1 + p2) * 0.5;
129  db = b.dot(p2 + p3) * 0.5;
130 
131  /*
132  We want to find a point c on each line, so:
133  [ a.x a.y ] [c.x] = [da]
134  [ b.x b.y ] [c.y] [db]
135 
136  We can invert the matrix to do that, and multiply it on the left of both
137  side. That yeilds a direct expression for center point c.
138  */
139 
140  num det = a.x * b.y - a.y * b.x;
141  if(fabs(det) < EPSILON) return(false);
142 
143  cen.x = ( b.y*da + -a.y*db) / det;
144  cen.y = (-b.x*da + a.x*db) / det;
145 
146  // use averaging to get a more numerically accurate radius
147  // rad = sqrt((sqdist(cen,p1) + sqdist(cen,p2) + sqdist(cen,p3)) / 3.0);
148  rad = dist(cen,p1);
149 
150  if(false){
151  printf("%f %f, %f %f\n",
152  a.dot(cen),da,
153  b.dot(cen),db);
154  }
155 
156  return(true);
157 }

169 template <class vector>
170 vector point_on_line(const vector x0,const vector x1,const vector p)
171 // returns nearest point on line through x0-x1 to point p
172 // Preconditions: x0!=x1
173 {
174  vector sx,sp,r;
175  double f,l;
176 
177  sx = x1 - x0;
178  sp = p - x0;
179 
180  f = dot(sx,sp);
181  l = sx.sqlength();
182 
183  // if line is degenerate, any point will do
184  if(l < EPSILON) return(p);
185 
186  // calculate point along line nearest to p
187  r = x0 + sx * (f / l);
188  return(r);
189 }

233 template <class vector>
234 double closest_point_time(const vector x1,const vector v1,
235  const vector x2,const vector v2)
236 // returns time of closest point of approach of two points
237 // moving along constant velocity vectors.
238 {
239  vector v = v1 - v2;
240  double sl = v.sqlength();
241  double t;
242 
243  if(sl < EPSILON) return(0.0); // parallel tracks, any time is ok.
244 
245  t = -v.dot(x1 - x2) / sl;
246  if(t < 0.0) return(0.0); // nearest time was in the past, now
247  // is closest point from now on.
248 
249  return(t);
250 }

260 template <class vector>
261 double distance_seg_to_seg(vector s1a,vector s1b,vector s2a,vector s2b)
262 // return distnace between segments s1a-s1b and s2a-s2b
263 {
264  vector dp;
265  vector u = s1b - s1a;
266  vector v = s2b - s2a;
267  vector w = s1a - s2a;
268  float a = dot(u,u); // always >= 0
269  float b = dot(u,v);
270  float c = dot(v,v); // always >= 0
271  float d = dot(u,w);
272  float e = dot(v,w);
273  float D = a*c - b*b; // always >= 0
274  float sc, sN, sD = D; // sc = sN / sD, default sD = D >= 0
275  float tc, tN, tD = D; // tc = tN / tD, default tD = D >= 0
276 
277  if(true){
278  printf("SegDist (%f,%f)-(%f,%f) to (%f,%f)-(%f,%f) a=%f b=%f\n",
279  V2COMP(s1a),V2COMP(s1b),V2COMP(s2a),V2COMP(s2b),
280  a,b);
281  }
282 
283  if((a < EPSILON) || (b < EPSILON)){
284  if((a < EPSILON) && (b < EPSILON)){
285  return(dist(s1a,s2a));
286  }else if(a < EPSILON){
287  return(distance_to_segment(s2a,s2b,s1a));
288  }else{
289  return(distance_to_segment(s1a,s1b,s2a));
290  }
291  }
292 
293  // compute the line parameters of the two closest points
294  if(D < EPSILON){ // the lines are almost parallel
295  sN = 0.0; // force using point P0 on segment S1
296  sD = 1.0; // to prevent possible division by 0.0 later
297  tN = e;
298  tD = c;
299  }else{ // get the closest points on the infinite lines
300  sN = (b*e - c*d);
301  tN = (a*e - b*d);
302  if(sN < 0){ // sc < 0 => the s=0 edge is visible
303  sN = 0.0;
304  tN = e;
305  tD = c;
306  }else if(sN > sD){ // sc > 1 => the s=1 edge is visible
307  sN = sD;
308  tN = e + b;
309  tD = c;
310  }
311  }
312 
313  if(tN < 0){ // tc < 0 => the t=0 edge is visible
314  tN = 0.0;
315  // recompute sc for this edge
316  if(-d < 0){
317  sN = 0.0;
318  }else if(-d > a){
319  sN = sD;
320  }else{
321  sN = -d;
322  sD = a;
323  }
324  }else if(tN > tD){ // tc > 1 => the t=1 edge is visible
325  tN = tD;
326  // recompute sc for this edge
327  if((-d + b) < 0){
328  sN = 0;
329  }else if((-d + b) > a){
330  sN = sD;
331  }else{
332  sN = (-d + b);
333  sD = a;
334  }
335  }
336 
337  if(false){
338  if(sD<1E-9 || tD<1E-9){
339  printf("seg_seg_degen D=%f sD=%f tD=%f\n",D,sD,tD);
340  printf(" SegDist (%f,%f)-(%f,%f) to (%f,%f)-(%f,%f)\n",
341  V2COMP(s1a),V2COMP(s1b),V2COMP(s2a),V2COMP(s2b));
342  }
343  }
344 
345  // finally do the division to get sc and tc
346  // sc = sN / sD;
347  // tc = tN / tD;
348  sc = (fabs(sN) < EPSILON)? 0.0 : sN / sD;
349  tc = (fabs(tN) < EPSILON)? 0.0 : tN / tD;
350 
351  // get the difference of the two closest points
352  dp = w + u*sc - v*tc; // = S1(sc) - S2(tc)
353 
354  return(dp.length()); // return the closest distance
355 }

374 template <class vector, class real>
375 real ray_sphere_intersect(vector rO, vector rV,vector sO,real sR)
376 // intersect a ray (r0+rV*t) with a sphere at s0 with radius sR
377 // returns: t, or -1 if no intersection
378 // NOTE: rV is assumed to be normalized
379 {
380  vector Q = sO - rO;
381  real c = Q.length();
382  real v = dot(Q,rV);
383  real d = sR*sR - (c*c - v*v);
384 
385  // If there was no intersection, return -1
386  if(d < 0.0) return(-1.0);
387 
388  // Return the distance to the [first] intersecting point
389  return(v - sqrt(d));
390 }

392 template <class vector, class real>
393 bool CircleTangentDir(vector cen,real rad,vector p,
394  vector &left,vector &right)
395 {
396  vector delta,basis;
397  vector tl,tr;
398  double d2,l,t;
399 
400  // l,r,d form a right triangle where l*l + rad*rad = d*d = d2
401  delta = cen - p;
402  d2 = delta.sqlength();
403 
404  t = d2 - rad*rad;
405  if(t < 0) return(false); // inside circle
406 
407  // set basis as sin/d,cos/d
408  l = sqrt(t);
409  basis.set(l/d2,rad/d2);
410 
411  // project to get left and right tangents
412  // resulting vectors are normalized
413  left = basis.project_out(delta);
414  basis.y = -basis.y;
415  right = basis.project_out(delta);
416 
417  return(true);
418 }
