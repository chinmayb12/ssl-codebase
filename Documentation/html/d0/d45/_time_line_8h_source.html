114  int findIdx ( TimeIndex idx ) const {
115  //perform a binary search for the index
116  int maxidx = list.size() - 1;
117  int minidx = 0;
118  int range;
119  int split;
120 
121  if ( maxidx < 0 ) return -1;
122 
123  while ( true ) {
124  range = maxidx - minidx;
125  split = ( range / 2 ) + minidx;
126  if ( range <= 1 ) return split;
127 
128  if ( * ( ( TimeIndex * ) ( list.at ( split ) ) ) == idx ) {
129  return split;
130  } else if ( * ( ( TimeIndex * ) ( list.at ( split ) ) ) < idx ) {
131  minidx = split;
132  } else {
133  maxidx = split;
134  }
135  }
