73 void PluginVisualize::DrawCameraImage(
74  FrameData* data, VisualizationFrame* vis_frame) {
75  //if converting entire image then blanking is not needed
76  const ColorFormat source_format = data->video.getColorFormat();
77  if (source_format == COLOR_RGB8) {
78  //plain copy of data
79  memcpy(vis_frame->data.getData(), data->video.getData(),
80  data->video.getNumBytes());
81  } else if (source_format==COLOR_YUV422_UYVY) {
82  Conversions::uyvy2rgb(
83  data->video.getData(),
84  reinterpret_cast<unsigned char*>(vis_frame->data.getData()),
85  data->video.getWidth(), data->video.getHeight());
86  } else {
87  //blank it:
88  vis_frame->data.fillBlack();
89  fprintf(stderr, "Unable to visualize color format: %s\n",
90  Colors::colorFormatToString(source_format).c_str());
91  fprintf(stderr, "Currently supported are rgb8 and yuv422 (UYVY).\n");
92  fprintf(stderr, "(Feel free to add more conversions to %s in %s).\n",
93  __FUNCTION__, __FILE__);
94  }
95  if (_v_greyscale->getBool() == true) {
96  unsigned int n = vis_frame->data.getNumPixels();
97  rgb * vis_ptr = vis_frame->data.getPixelData();
98  rgb color;
99  for (unsigned int i = 0; i < n; i++) {
100  color = vis_ptr[i];
101  color.r = color.g = color.b = ((color.r + color.g + color.b)/3);
102  vis_ptr[i] = color;
103  }
104  }
105 }

207 void PluginVisualize::DrawSobelImage(
208  FrameData* data, VisualizationFrame* vis_frame) {
209  if (edge_image == 0) {
210  edge_image =
211  new greyImage(data->video.getWidth(),data->video.getHeight());
212  temp_grey_image =
213  new greyImage(data->video.getWidth(),data->video.getHeight());
214  }
215  Images::convert(vis_frame->data, *temp_grey_image);
216  // Draw sobel image: Contrast towards more brightness is painted white,
217  // Contrast towards more darkness is painted green
218  for(int y=1; y<temp_grey_image->getHeight()-1; ++y)
219  {
220  for(int x=1; x<temp_grey_image->getWidth()-1; ++x)
221  {
222  int colVB,colVD,colHB,colHD;
223  colVB = Sobel::verticalBrighter(*temp_grey_image,x,y,30);
224  colVD = Sobel::verticalDarker(*temp_grey_image,x,y,30);
225  colHB = Sobel::horizontalBrighter(*temp_grey_image,x,y,30);
226  colHD = Sobel::horizontalDarker(*temp_grey_image,x,y,30);
227  int dMax = colVD > colHD ? colVD : colHD;
228  int bMax = colVB > colHB ? colVB : colHB;
229  grey col;
230  if (dMax > bMax)
231  col.v = 1;
232  else if (bMax > dMax)
233  col.v = 2;
234  else
235  col.v = 0;
236  edge_image->setPixel(x,y,col);
237  }
238  }
239  unsigned int n = edge_image->getNumPixels();
240  rgb * vis_ptr = vis_frame->data.getPixelData();
241  grey * edge_ptr = edge_image->getPixelData();
242  rgb color;
243  for (unsigned int i=0;i<n;i++) {
244  unsigned char col = edge_ptr[i].v;
245  if (col == 0)
246  {
247  color.r=color.g=color.b=col;
248  }
249  else if (col == 1)
250  {
251  color.r=0; color.g=255; color.b=0;
252  }
253  else if (col == 2)
254  {
255  color.r=255; color.g=255; color.b=255;
256  }
257  vis_ptr[i]=color;
258  }
259 }

279 void PluginVisualize::DrawDetectedEdges(
280  FrameData* data, VisualizationFrame* vis_frame) {
281  // The edges:
282  const rgb edge_draw_color = RGB::Red;
283  for (size_t ls = 0; ls < camera_parameters.calibrationSegments.size(); ++ls) {
284  const CameraParameters::CalibrationData& segment =
285  camera_parameters.calibrationSegments[ls];
286  for(unsigned int edge=0; edge<segment.imgPts.size(); ++edge) {
287  if (!(segment.imgPts[edge].second)) continue;
288  const GVector::vector2d<double>& image_point = segment.imgPts[edge].first;
289  vis_frame->data.drawBox(
290  image_point.x - 5, image_point.y - 5, 11, 11, edge_draw_color);
291  const double alpha = segment.alphas[edge];
292  if (segment.straightLine) {
293  const GVector::vector3d<double> field_point =
294  alpha * segment.p1 + (1.0 - alpha) * segment.p2;
295  const GVector::vector3d<double> field_tangent =
296  (segment.p2 - segment.p1).norm();
297  DrawEdgeTangent(image_point, field_point, field_tangent, vis_frame,
298  edge_draw_color);
299  } else {
300  const double angle =
301  alpha * segment.theta1 + (1.0 - alpha) * segment.theta2;
302  const GVector::vector3d<double> field_radius_vector(
303  cos(angle), sin(angle), 0.0);
304  const GVector::vector3d<double> field_point =
305  segment.center + segment.radius * field_radius_vector;
306  const GVector::vector3d<double> field_tangent(
307  -sin(angle), cos(angle), 0.0);
308  DrawEdgeTangent(image_point, field_point, field_tangent, vis_frame,
309  edge_draw_color);
310  }
311  }
312  }
313  DrawSearchCorridors(data, vis_frame);
314 }

316 void PluginVisualize::DrawSearchCorridors(
317  FrameData* data, VisualizationFrame* vis_frame) {
318  static const int steps_per_line = 20;
319  const double half_corridor_width = camera_parameters.
320  additional_calibration_information->line_search_corridor_width->
321  getDouble() * 0.5;
322  real_field.field_markings_mutex.lockForRead();
323  for (size_t i = 0; i < real_field.field_lines.size(); ++i) {
324  const FieldLine& line_segment =
325  *(real_field.field_lines[i]);
326  const GVector::vector3d<double> p1(
327  line_segment.p1_x->getDouble(), line_segment.p1_y->getDouble(), 0.0);
328  const GVector::vector3d<double> p2(
329  line_segment.p2_x->getDouble(), line_segment.p2_y->getDouble(), 0.0);
330  const GVector::vector3d<double> line_dir = (p2 - p1).norm();
331  const GVector::vector3d<double> line_perp(-line_dir.y, line_dir.x, 0.0);
332  drawFieldLine(p1 + half_corridor_width * line_perp,
333  p2 + half_corridor_width * line_perp,
334  steps_per_line, vis_frame, 180, 180, 255);
335  drawFieldLine(p1 - half_corridor_width * line_perp,
336  p2 - half_corridor_width * line_perp,
337  steps_per_line, vis_frame, 180, 180, 255);
338  }
339  for (size_t i = 0; i < real_field.field_arcs.size(); ++i) {
340  const FieldCircularArc& arc = *(real_field.field_arcs[i]);
341  const GVector::vector3d<double> center(
342  arc.center_x->getDouble(), arc.center_y->getDouble(), 0.0);
343  drawFieldArc(center, arc.radius->getDouble() + half_corridor_width,
344  arc.a1->getDouble(), arc.a2->getDouble(), steps_per_line,
345  vis_frame, 180, 180, 255);
346  drawFieldArc(center,
347  max(0.0, arc.radius->getDouble() - half_corridor_width),
348  arc.a1->getDouble(), arc.a2->getDouble(), steps_per_line,
349  vis_frame, 180, 180, 255);
350  }
351  real_field.field_markings_mutex.unlock();
352 }

354 ProcessResult PluginVisualize::process(
355  FrameData* data, RenderOptions* options) {
356  if (data == 0) return ProcessingFailed;
357 
358  VisualizationFrame* vis_frame =
359  reinterpret_cast<VisualizationFrame*>(data->map.get("vis_frame"));
360  if (vis_frame == 0) {
361  vis_frame = reinterpret_cast<VisualizationFrame*>(
362  data->map.insert("vis_frame",new VisualizationFrame()));
363  }
364 
365  if (_v_enabled->getBool()) {
366  //check video data...
367  if (data->video.getWidth() == 0 || data->video.getHeight()==0) {
368  //there is no valid video data
369  //mark visualization data as invalid
370  vis_frame->valid = false;
371  return ProcessingOk;
372  } else {
373  //allocate visualization frame accordingly:
374  vis_frame->data.allocate(data->video.getWidth(), data->video.getHeight());
375  }
376 
377  // Draw camera image
378  if (_v_image->getBool()) {
379  DrawCameraImage(data, vis_frame);
380  } else {
381  vis_frame->data.fillBlack();
382  }
383 
384  // Draw color-thresholded image.
385  if (_v_thresholded->getBool()) {
386  DrawThresholdedImage(data, vis_frame);
387  }
388 
389  //draw blob finding results:
390  if (_v_blobs->getBool()) {
391  DrawBlobs(data, vis_frame);
392  }
393 
394  // Camera calibration
395  if (_v_camera_calibration->getBool()) {
396  DrawCameraCalibration(data, vis_frame);
397  }
398 
399  // Result of camera calibration, draws field to image
400  if (_v_calibration_result->getBool()) {
401  DrawCalibrationResult(data, vis_frame);
402  }
403 
404  // Test edge detection for calibration
405  if (_v_complete_sobel->getBool()) {
406  DrawSobelImage(data, vis_frame);
407  }
408 
409  // Result of edge detection for second calibration step
410  if (_v_detected_edges->getBool()) {
411  DrawDetectedEdges(data, vis_frame);
412  }
413  vis_frame->valid = true;
414  } else {
415  vis_frame->valid = false;
416  }
417  return ProcessingOk;
418 }

424 void PluginVisualize::drawFieldArc(
425  const GVector::vector3d<double>& center,
426  double radius, double theta1, double theta2, int steps,
427  VisualizationFrame* vis_frame,
428  unsigned char r, unsigned char g, unsigned char b) {
429  const double delta = (theta2 - theta1) / static_cast<double>(steps);
430  GVector::vector2d<double> lastInImage(0.0, 0.0);
431  GVector::vector3d<double> lastInWorld = center +
432  radius * GVector::vector3d<double>(cos(theta1), sin(theta1), 0.0);
433  camera_parameters.field2image(lastInWorld, lastInImage);
434  for (int i = 1; i <= steps; ++i) {
435  const double theta = theta1 + static_cast<double>(i) * delta;
436  GVector::vector3d<double> nextInWorld = center +
437  radius * GVector::vector3d<double>(cos(theta), sin(theta), 0.0);
438  GVector::vector2d<double> nextInImage(0.0, 0.0);
439  camera_parameters.field2image(nextInWorld, nextInImage);
440  rgb draw_color;
441  draw_color.set(r,g,b);
442  vis_frame->data.drawFatLine(
443  lastInImage.x,lastInImage.y,nextInImage.x,nextInImage.y,draw_color);
444  lastInWorld = nextInWorld;
445  lastInImage = nextInImage;
446  }
