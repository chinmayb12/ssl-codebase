300 void PluginDVR::slotMovieLoad() {
301  lock();
302  // Do not use native dialog since some platforms like Windows XP and
303  // Kubuntu 12.04 have very slow directory listing dialogs. The QT version is
304  // pretty responsive.
305  QString dirstr = QFileDialog::getExistingDirectory(
306  0,"Select Directory to Load", "",
307  QFileDialog::DontUseNativeDialog | QFileDialog::ShowDirsOnly);
308  if (dirstr!="") {
309 
310  int fnum=0;
311  stream.clear();
312  QDir dir(dirstr);
313  QFileInfoList list = dir.entryInfoList ( QDir::Files | QDir::Readable, QDir::Name );
314  QProgressDialog * dlg = new QProgressDialog("Loading Movie from Files...","Cancel", 1,list.size());
315  dlg->setWindowModality(Qt::WindowModal);
316  for (int i = 0 ; i < list.size(); i ++) {
317  dlg->setValue(i+1);
318  QFileInfo info = list[i];
319  if (info.suffix().compare("png",Qt::CaseInsensitive) == 0 ||
320  info.suffix().compare("bmp",Qt::CaseInsensitive) == 0 ||
321  info.suffix().compare("jpg",Qt::CaseInsensitive) == 0 ||
322  info.suffix().compare("jpeg",Qt::CaseInsensitive) == 0) {
323  int w,h;
324  rgb * data = ImageIO::readRGB(w,h,info.filePath().toStdString().c_str());
325  FrameData fdata;
326  fdata.time = 0.0;
327  fdata.number = fnum;
328  fnum++;
329  fdata.video.setColorFormat(COLOR_RGB8);
330  fdata.video.setData((unsigned char*)data);
331  fdata.video.setWidth(w);
332  fdata.video.setHeight(h);
333  stream.appendFrame(&fdata,false);
334  delete[] data;
335  }
336  if (dlg->wasCanceled()) break;
337  }
338  delete dlg;
339  }
340  unlock();
341 }

343 void PluginDVR::slotMovieSave() {
344  lock();
345  QString dir = QFileDialog::getExistingDirectory(0,"Select Directory to Save");
346  rgbImage output;
347  QProgressDialog * dlg = new QProgressDialog("Saving Movie to PNG Files...","Cancel", 1,stream.getFrameCount());
348  dlg->setWindowModality(Qt::WindowModal);
349 
350  if (dir!="") {
351  for (int i = 0; i < stream.getFrameCount(); i++) {
352  DVRFrame * f = stream.getFrame(i);
353  dlg->setValue(i+1);
354  if (f!=0) {
355  ColorFormat fmt=f->video.getColorFormat();
356  output.allocate(f->video.getWidth(),f->video.getHeight());
357  if (fmt==COLOR_YUV422_UYVY) {
358  Conversions::uyvy2rgb(f->video.getData(),output.getData(),f->video.getWidth(),f->video.getHeight());
359  } else if (fmt==COLOR_RGB8) {
360  memcpy(output.getData(),f->video.getData(),f->video.getNumBytes());
361  } else {
362  output.allocate(0,0);
363  }
364  if (output.getNumBytes() > 0) {
365  //write file:
366  QString num = QString::number(i);
367  num = "00000" + num;
368  num = num.right(5);
369  QString filename = dir + "/" + num + ".png";
370  output.save(filename.toStdString());
371  }
372  }
373  if (dlg->wasCanceled()) break;
374  }
375  }
376  delete dlg;
377  unlock();
378 }

417 ProcessResult PluginDVR::process(FrameData * data, RenderOptions * options) {
418  (void)options;
419  QString status;
420  QString stream_info;
421  stream_info = "";
422  status = "";
423  bool is_recording=w->btn_rec_rec->isChecked();
424  //capture data:
425  if (mode==DVRModePause) {
426  if (trigger_pause_refresh) {
427  pause_frame.getFromFrameData(data);
428  trigger_pause_refresh=false;
429  }
430  status = "Pausing.";
431  } else if (mode==DVRModeRecord) {
432  if (is_recording) {
433  stream.setLimit(_max_frames->getInt());
434  stream.appendFrame(data,_shift_on_exceed->getBool());
435  status = "Recording (Frame " + QString::number(stream.getFrameCount()) + ").";
436  if (stream.getFrameCount() >= stream.getLimit()) {
437  if (_shift_on_exceed->getBool()) {
438  status = status + " Past Max Frame Limit! Now Shift-Recoding!";
439  } else {
440  status = status + " Past Max Frame Limit! Dropping Frames!";
441  }
442  }
443  } else {
444  status = "Recorded " + QString::number(stream.getFrameCount()) + " Frame(s)";
445  }
446  } else {
447  status = "Live Pass-Through.";
448  }
449 
450  if (mode==DVRModeRecord) {
451  //move stream:
452  double t=GetTimeSec();
453  double advance_passed=t-advance_last_t;
454  double advance_fps=(double)w->jog->offset() * 20.0;
455 
456  int frames_advance_int=0;
457  if (advance_fps != 0.0) {
458  double advance_period=1.0/advance_fps;
459  float frames_behind=advance_passed/advance_period;
460  if (fabs(frames_behind) > 1.0) {
461  frames_advance_int=(int)frames_behind;
462  }
463  }
464  if (frames_advance_int!=0) {
465  w->btn_seek_pause->setChecked(true);
466  seek_mode=SeekModePause;
467  }
468 
469 
470  if (seek_mode==SeekModePlay) {
471  stream.advance(1,w->btn_seek_wrap->isChecked());
472  } else if (seek_mode==SeekModePause) {
473  if (frames_advance_int!=0) {
474  stream.advance(frames_advance_int,w->btn_seek_wrap->isChecked());
475  advance_last_t=t;
476  }
477  }
478  if (stream.getFrameCount() <= 0 || seek_mode==SeekModeLive) {
479  stream_info = "Showing Live-View.";
480  } else {
481  double percent = 100.0 * ((double)stream.getCurrentFrameIndex()+1) / ((double)(stream.getFrameCount()));
482  if (seek_mode==SeekModePause) {
483  stream_info = "Paused Playback at Frame " + QString::number(stream.getCurrentFrameIndex()+1) + "/" + QString::number(stream.getFrameCount()) + " (" + QString::number(percent,'f',2) + "%).";
484  } else {
485  stream_info = "Running Playback at Frame " + QString::number(stream.getCurrentFrameIndex()+1) + "/" + QString::number(stream.getFrameCount()) + " (" + QString::number(percent,'f',2) + "%).";
486  }
487  }
488  }
489 
490  //output data:
491  if (mode==DVRModePause) {
492  data->video.deepCopyFromRawImage(pause_frame.video,false);
493  } else if (mode==DVRModeRecord) {
494  if (seek_mode!=SeekModeLive && stream.getFrameCount() > 0) {
495  DVRFrame * f = stream.getCurrentFrame();
496  if (f!=0) {
497  data->video.deepCopyFromRawImage(f->video,false);
498  }
499  }
500  }
501 
502  //update GUI:
503  w->label_info->setText(status + "\n" + stream_info);
504 
505  return ProcessingOk;
506 }

573 void DVRStream::advance(int frames, bool wrap) {
574  if (getFrameCount() == 0) return;
575  int new_current=current+frames;
576  int max_idx=getFrameCount()-1;
577  if (new_current < 0) {
578  if (wrap) {
579  new_current=mod(new_current,getFrameCount());
580  } else {
581  new_current=0;
582  }
583  }
584  if (new_current > max_idx) {
585  if (wrap) {
586  new_current=mod(new_current,getFrameCount());
587  } else {
588  new_current=max_idx;
589  }
590  }
591  current=new_current;
