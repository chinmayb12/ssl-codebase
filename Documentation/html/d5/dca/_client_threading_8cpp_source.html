54 int ViewUpdateThread::execute()
55 {
56  Log_Frame* log_frame;
57  SSL_DetectionFrame detection;
58  if ( client.receive ( packet ) && !play)
59  {
60  //see if the packet contains a robot detection frame:
61  if ( packet.has_detection() )
62  {
63  detection = packet.detection();
64  int balls_n = detection.balls_size();
65  //Ball info:
66  QVector<QPointF> balls;
67  for ( int i = 0; i < balls_n; i++ )
68  {
69  QPointF p;
70  SSL_DetectionBall ball = detection.balls ( i );
71  if ( ball.confidence() > 0.0 )
72  {
73  p.setX ( ball.x() );
74  p.setY ( ball.y() );
75  balls.push_back ( p );
76  }
77  }
78  drawMutex->lock();
79  soccerView->UpdateBalls ( balls,detection.camera_id() );
80  //Robot info:
81  soccerView->UpdateRobots ( detection );
82  drawMutex->unlock();
83  }
84  //see if packet contains geometry data:
85  if ( packet.has_geometry() )
86  {
87  drawMutex->lock();
88  const SSL_GeometryData & geom = packet.geometry();
89  const SSL_GeometryFieldSize & field = geom.field();
90  soccerView->LoadFieldGeometry ( ( SSL_GeometryFieldSize& ) field );
91  drawMutex->unlock();
92  }
93  }
94 
95  if(!play && log_control->get_current_frame() != 0)
96  end_play_record();
97 
98  //Play logfile
99  if(play)
100  {
101  if(log_control->get_next_frame() < 0)
102  end_play_record();
103  else
104  {
105  //get next frame
106  log_frame = logs.mutable_log(log_control->get_current_frame());
107  SSL_DetectionFrame* det_frame = log_frame->mutable_frame();
108  detection = *det_frame;
109 
110  //process frame
111  int balls_n = detection.balls_size();
112  //Ball info:
113  QVector<QPointF> balls;
114  for ( int i = 0; i < balls_n; i++ )
115  {
116  QPointF p;
117  SSL_DetectionBall ball = detection.balls ( i );
118  if ( ball.confidence() > 0.0 )
119  {
120  p.setX ( ball.x() );
121  p.setY ( ball.y() );
122  balls.push_back ( p );
123  }
124  }
125  drawMutex->lock();
126  soccerView->UpdateBalls ( balls,detection.camera_id() );
127  //Robot info:
128  soccerView->UpdateRobots ( detection );
129  drawMutex->unlock();
130  }
131 
132  emit update_frame(log_control->get_current_frame());
133 
134  //distance between frames in ms
135  if(log_control->get_play_speed() == 0)
136  return 10;
137 
138  //calculate distance between frames
139  if(!(log_control->get_prop_next_frame() < 0))
140  {
141  double old_time = detection.t_capture();
142  double new_time = logs.mutable_log(log_control->get_prop_next_frame())->mutable_frame()->t_capture();
143  double timediff = new_time - old_time;
144  return ((timediff * 1000) / log_control->get_play_speed());
145  }
146 
147  }
148 
149  //distance between frames in ms, see http://xkcd.com/221/
150  return 4;
151 }

174 int ViewUpdateThread::start_play_record()
175 {
176  //change fileName into directory
177  if(fileName != QDir::homePath())
178  {
179  int last_slash = 0;
180  for(int i = 0; i < fileName.length(); i++)
181  {
182  if(fileName.at(i).toAscii() == (QChar('/')))
183  last_slash = i;
184  }
185  fileName.remove(last_slash, fileName.length()-last_slash);
186  }
187 
188  //What data shall I read?
189  fileName = QFileDialog::getOpenFileName((QWidget*)this->parent(), tr("Open Logfile"), fileName, tr("Log Files (*.log)"));
190  std::cout << "fileName: " << fileName.toAscii().constData() << std::endl;
191 
192  // Read the existing log.
193  std::fstream input(fileName.toAscii().constData(), std::ios::in | std::ios::binary);
194  if (!input)
195  {
196  std::cout << fileName.toAscii().constData() << ": File not found." << std::endl;
197  return -1;
198  }
199  else if (!logs.ParseFromIstream(&input))
200  {
201  std::cout << "Failed to parse Logfile." << std::endl;
202  return -1;
203  }
204 
205  std::cout << "File successfully loaded" << std::endl;
206  if(logs.IsInitialized() && logs.log_size() > 0 && logs.log(0).IsInitialized())
207  {
208  std::cout << "Logfilegröße: " << logs.log_size() << std::endl;
209  std::cout << "Start Command: " << logs.mutable_log(0)->refbox_cmd() << std::endl;
210  log_control->reset(logs.log_size());
211  emit log_size(logs.log_size());
212  //initializeSlider(int min, int max, int singleStep, int pageStep, int tickInterval)
213  emit initializeSlider(0, logs.log_size(), 1, 100, 1800);
214  emit showLogControl(true);
215  }
216  else
217  {
218  std::cout << "Logfile seems to be empty or damaged" << std::endl;
219  return -1;
220  }
221  input.close();
222  emit change_play_button(" End Play ");
223  return 0;
224 }
