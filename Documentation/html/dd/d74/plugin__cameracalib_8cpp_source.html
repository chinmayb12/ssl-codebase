57 void PluginCameraCalibration::detectEdges(FrameData* data) {
58  double point_separation(camera_parameters.additional_calibration_information->
59  pointSeparation->getDouble());
60  // Reset list:
61  camera_parameters.calibrationSegments.clear();
62  // Get a greyscale image:
63  if(grey_image == 0) {
64  grey_image = new greyImage(data->video.getWidth(),data->video.getHeight());
65  rgb_image = new rgbImage(data->video.getWidth(),data->video.getHeight());
66  }
67  if (data->video.getColorFormat()==COLOR_YUV422_UYVY) {
68  Conversions::uyvy2rgb(
69  data->video.getData(),
70  reinterpret_cast<unsigned char*>(rgb_image->getData()),
71  data->video.getWidth(),data->video.getHeight());
72  Images::convert(*rgb_image, *grey_image);
73  } else if (data->video.getColorFormat()==COLOR_RGB8) {
74  Images::convert(data->video, *grey_image);
75  } else {
76  fprintf(stderr, "ColorThresholding needs YUV422 or RGB8 as input image, "
77  "but found: %s\n",
78  Colors::colorFormatToString(data->video.getColorFormat()).c_str());
79  return;
80  }
81 
82  field.field_markings_mutex.lockForRead();
83  for (size_t i = 0; i < field.field_lines.size(); ++i) {
84  const FieldLine& line = *(field.field_lines[i]);
85  const GVector::vector3d<double> p1(
86  line.p1_x->getDouble(), line.p1_y->getDouble(), 0.0);
87  const GVector::vector3d<double> p2(
88  line.p2_x->getDouble(), line.p2_y->getDouble(), 0.0);
89  detectEdgesOnSingleLine(
90  p1, p2, line.thickness->getDouble(), point_separation);
91  }
92 
93  for (size_t i = 0; i < field.field_arcs.size(); ++i) {
94  const FieldCircularArc& arc = *(field.field_arcs[i]);
95  const GVector::vector3d<double> center(
96  arc.center_x->getDouble(), arc.center_y->getDouble(), 0.0);
97  detectEdgesOnSingleArc(center, arc.radius->getDouble(), arc.a1->getDouble(),
98  arc.a2->getDouble(), arc.thickness->getDouble(), point_separation);
99  }
100  field.field_markings_mutex.unlock();
101 }

140 void PluginCameraCalibration::detectEdgesOnSingleLine(
141  const GVector::vector3d<double>& p1,
142  const GVector::vector3d<double>& p2,
143  double thickness, double point_separation) {
144  static const int kSobelThreshold = 20;
145  const double search_distance =
146  camera_parameters.additional_calibration_information->
147  line_search_corridor_width->getDouble() / 2.0;
148  const double image_boundary =
149  camera_parameters.additional_calibration_information->
150  image_boundary->getDouble();
151  const double max_feature_distance =
152  camera_parameters.additional_calibration_information->
153  max_feature_distance->getDouble();
154  CameraParameters::CalibrationData calibration_data;
155  calibration_data.straightLine = true;
156  calibration_data.p1 = p1;
157  calibration_data.p2 = p2;
158  const double line_length = (p2 - p1).length();
159  const GVector::vector3d<double> line_dir = (p2 - p1) / line_length;
160  const GVector::vector3d<double> line_perp(-line_dir.y, line_dir.x, 0.0);
161  const int num_points = floor(line_length / point_separation - 1.0);
162  for (int i = 1; i <= num_points; ++i) {
163  const double alpha =
164  1.0 - static_cast<double>(i) / (static_cast<double>(num_points) + 1.0);
165  const GVector::vector3d<double> p_world = alpha * p1 + (1.0 - alpha) * p2;
166  GVector::vector2d<double> p_image(0.0, 0.0);
167  camera_parameters.field2image(p_world, p_image);
168  if (p_image.x < image_boundary ||
169  p_image.x > grey_image->getWidth() - image_boundary ||
170  p_image.y < image_boundary ||
171  p_image.y > grey_image->getHeight() - image_boundary) {
172  // This edge feature is outside the image boundary, so ignore it since
173  // the edge detection might not be accurate.
174  continue;
175  }
176  if ((p_world - camera_parameters.getWorldLocation()).length() >
177  max_feature_distance) {
178  // This edge feature too far away from the camera, so ignore it since the
179  // line feature is likely to be too small to detect accurately.
180  continue;
181  }
182  const GVector::vector3d<double> start_world =
183  p_world + line_perp * (0.5 * thickness + search_distance);
184  const GVector::vector3d<double> end_world =
185  p_world - line_perp * (0.5 * thickness + search_distance);
186  GVector::vector2d<double> start_image(0.0, 0.0), end_image(0.0, 0.0);
187  camera_parameters.field2image(start_world, start_image);
188  camera_parameters.field2image(end_world, end_image);
189  sanitizeSobel(grey_image, start_image);
190  sanitizeSobel(grey_image, end_image);
191 
192  GVector::vector2d<double> point_image;
193  bool center_found = false;
194  Sobel::centerOfLine(
195  *grey_image, start_image.x, end_image.x, start_image.y,
196  end_image.y, point_image, center_found, kSobelThreshold);
197  calibration_data.imgPts.push_back(std::make_pair(point_image,center_found));
198  calibration_data.alphas.push_back(alpha);
199  }
200  if (calibration_data.imgPts.size() > 2) {
201  // Need at least two points to uniquely define a line segment.
202  camera_parameters.calibrationSegments.push_back(calibration_data);
203  }
204 }

206 void PluginCameraCalibration::detectEdgesOnSingleArc(
207  const GVector::vector3d<double>& center, double radius, double theta1,
208  double theta2, double thickness, double point_separation) {
209  static const int kSobelThreshold = 20;
210  const double search_distance =
211  camera_parameters.additional_calibration_information->
212  line_search_corridor_width->getDouble() / 2.0;
213  const double image_boundary =
214  camera_parameters.additional_calibration_information->
215  image_boundary->getDouble();
216  const double max_feature_distance =
217  camera_parameters.additional_calibration_information->
218  max_feature_distance->getDouble();
219  CameraParameters::CalibrationData calibration_data;
220  calibration_data.straightLine = false;
221  calibration_data.radius = radius;
222  calibration_data.center = center;
223  calibration_data.theta1 = theta1;
224  calibration_data.theta2 = theta2;
225  const int num_points =
226  floor((theta2 - theta1) * radius / point_separation - 1.0);
227  for (int i = 1; i <= num_points; ++i) {
228  const double alpha =
229  1.0 - static_cast<double>(i) / (static_cast<double>(num_points) + 1.0);
230  const double theta = alpha * theta1 + (1.0 - alpha) * theta2;
231  const GVector::vector3d<double> radius_vector(cos(theta), sin(theta), 0.0);
232  const GVector::vector3d<double> p_world = center + radius * radius_vector;
233  GVector::vector2d<double> p_image(0.0, 0.0);
234  camera_parameters.field2image(p_world, p_image);
235  if (p_image.x < image_boundary ||
236  p_image.x > grey_image->getWidth() - image_boundary ||
237  p_image.y < image_boundary ||
238  p_image.y > grey_image->getHeight() - image_boundary) {
239  // This edge feature is outside the image boundary, so ignore it since
240  // the edge detection might not be accurate.
241  continue;
242  }
243  if ((p_world - camera_parameters.getWorldLocation()).length() >
244  max_feature_distance) {
245  // This edge feature too far away from the camera, so ignore it since the
246  // line feature is likely to be too small to detect accurately.
247  continue;
248  }
249  const GVector::vector3d<double> start_world =
250  center + (radius - 0.5 * thickness - search_distance) * radius_vector;
251  const GVector::vector3d<double> end_world =
252  center + (radius + 0.5 * thickness + search_distance) * radius_vector;
253  GVector::vector2d<double> start_image(0.0, 0.0), end_image(0.0, 0.0);
254  camera_parameters.field2image(start_world, start_image);
255  camera_parameters.field2image(end_world, end_image);
256  sanitizeSobel(grey_image, start_image);
257  sanitizeSobel(grey_image, end_image);
258 
259  GVector::vector2d<double> point_image;
260  bool center_found = false;
261  Sobel::centerOfLine(
262  *grey_image, start_image.x, end_image.x, start_image.y,
263  end_image.y, point_image, center_found, kSobelThreshold);
264  calibration_data.imgPts.push_back(std::make_pair(point_image,center_found));
265  calibration_data.alphas.push_back(alpha);
266  }
267  if (calibration_data.imgPts.size() > 3) {
268  // Need at least three points to uniquely define a circular arc.
269  camera_parameters.calibrationSegments.push_back(calibration_data);
270  }
271 }

284 void PluginCameraCalibration::mousePressEvent ( QMouseEvent * event, pixelloc loc )
285 {
286  QTabWidget* tabw = (QTabWidget*) ccw->parentWidget()->parentWidget();
287  double drag_threshold = 20; //in px
288  if (tabw->currentWidget() == ccw && (event->buttons() & Qt::LeftButton)!=0) {
289  drag_x = 0;
290  drag_y = 0;
291  for (int i = 0;
292  i < CameraParameters::AdditionalCalibrationInformation::kNumControlPoints;
293  ++i) {
294  const double x_diff =
295  camera_parameters.additional_calibration_information->
296  control_point_image_xs[i]->getDouble() - loc.x;
297  const double y_diff =
298  camera_parameters.additional_calibration_information->
299  control_point_image_ys[i]->getDouble() - loc.y;
300  if (sqrt(x_diff*x_diff + y_diff*y_diff) < drag_threshold) {
301  drag_x = camera_parameters.additional_calibration_information->
302  control_point_image_xs[i];
303  drag_y = camera_parameters.additional_calibration_information->
304  control_point_image_ys[i];
305  break;
306  }
307  }
308  if (drag_x != 0 && drag_y != 0)
309  {
310  event->accept();
311  doing_drag = true;
312  }
313  else
314  event->ignore();
315  } else {
316  event->ignore();
317  }
