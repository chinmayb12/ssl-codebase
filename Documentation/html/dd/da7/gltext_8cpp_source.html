134 {

141 
142  switch(hAlign){
143  case LeftAligned:{
144  //Normal rendering will achieve this!
145  break;
146  }
147  case RightAligned:{
148  glTranslated(-getWidth(str),0,0);
149  break;
150  }
151  case CenterAligned:{
152  glTranslated(-0.5*getWidth(str),0,0);

154  }
155  }
156  switch(vAlign){
157  case BottomAligned:{
158  glTranslated(0.0,getDescent(str),0.0);
159  break;
160  }
161  case TopAligned:{

164  }
165  case MedianAligned:{
166  //Normal rendering will achieve this!
167  break;
168  }
169  case MiddleAligned:{
170  glTranslated(0.0,-0.5*getHeight(str),0.0);
171  break;
172  }
173  }
174  vector2d textDir;
175  textDir.heading(0);
176  double d = 0.0;
177  while(*str>0){
178  drawGlyph(*str);
179  d = characterSpacing + 0.5*getWidth(*str);
180  str++;
181  if((*str)>0){
182  d += 0.5*getWidth(*str);
183  glTranslated(d*textDir.x,d*textDir.y,0.0);
184  }
185  }
186  glPopMatrix();
187 }

189 void GLText::initializeGlyph(char ch)
190 {
191  Glyph glyph;
192  QPainterPath path;
193  if(debugTesselation) printf("Adding glyph %c\n",ch);
194  path.addText(0,0,font,QString((QChar)ch));
195  QList<QPolygonF> polygons = path.toSubpathPolygons();
196  if(debugTesselation){
197  printf("%d Sub-Polygons\n",polygons.size());
198  printf("Poly has %d vertices:\n",polygons.size());
199  }
200  int numVertices = 0;
201  double minX=DBL_MAX, minY=DBL_MAX, maxX=-DBL_MAX, maxY=-DBL_MAX;
202  for(int i=0; i<polygons.size(); i++){
203  if(debugTesselation) printf("Sub-Polygon %d:\n",i);
204  numVertices += polygons[i].size();
205 
206  for(int k=0; k<polygons[i].size(); k++){
207  if(debugTesselation) printf("%8.3f,%8.3f\n",polygons[i][k].x(),polygons[i][k].y());
208  minX = min(minX, polygons[i][k].x());
209  maxX = max(maxX, polygons[i][k].x());
210  minY = min(minY, polygons[i][k].y());
211  maxY = max(maxY, polygons[i][k].y());
212  }
213  }
214  glyph.ascent = fabs(minY)/FontRenderSize;
215  glyph.descent = fabs(maxY)/FontRenderSize;
216 
217  glyph.height = (maxY - minY)/FontRenderSize;
218  glyph.width = (maxX - minX)/FontRenderSize;
219 
220  if(debugTesselation) printf("numVertices: %d\n",numVertices);
221  GLdouble vertices[numVertices][3];
222  int j=0;
223  for(int i=0; i<polygons.size(); i++){
224  for(int k=0; k<polygons[i].size(); k++){
225  vertices[j][0] = polygons[i][k].x()/FontRenderSize;
226  vertices[j][1] = -polygons[i][k].y()/FontRenderSize;
227  vertices[j][2] = 9;
228  j++;
229  }
230  }
231 
232  GLUtesselator* tess = gluNewTess();
233  gluTessCallback(tess, GLU_TESS_BEGIN, (_GLUfuncptr) tessBeginCB);
234  gluTessCallback(tess, GLU_TESS_END, (_GLUfuncptr) tessEndCB);
235  gluTessCallback(tess, GLU_TESS_ERROR, (_GLUfuncptr) tessErrorCB);
236  gluTessCallback(tess, GLU_TESS_VERTEX, (_GLUfuncptr) tessVertexCB);
237 
238  glMatrixMode(GL_MODELVIEW);
239  glPushMatrix();
240  glLoadIdentity();
241 
242  glyph.displayListID = glGenLists(1);
243  if(glyph.displayListID==GL_INVALID_VALUE){
244  printf("Unable to create display list!\n");
245  exit(1);
246  }
247  glNewList(glyph.displayListID, GL_COMPILE);
248  gluTessBeginPolygon(tess, 0);
249  j=0;
250  for(int i=0; i<polygons.size(); i++){
251  gluTessBeginContour(tess);
252  for(int k=0; k<polygons[i].size(); k++){
253  gluTessVertex(tess, vertices[j], vertices[j]);
254  j++;
255  }
256  gluTessEndContour(tess);
257  }
258  gluTessEndPolygon(tess);
259  gluDeleteTess(tess);
260  glEndList();
261  glPopMatrix();
262  glyph.compiled = true;
263  glyphs[ch] = glyph;
264 }
