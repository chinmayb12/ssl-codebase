95  void VarNotifier::internalNonMutexedAddItem(VarType * item, VarNotificationType notification_type) {
96  if (item==0) return;
97  QHash<VarType *, VarNotificationType>::iterator iter;
98  iter = senders.find(item);
99  if (iter != senders.constEnd()) {
100  if (iter.value() == notification_type) {
101  //this exact item already exists.
102  return;
103  } else {
104  //item exists, but with wrong notification type:
105  //fix it:
106  disconnect(item, 0, this, SLOT(changeSlot(VarType *)));
107  iter.value() = notification_type;
108  }
109  } else {
110  iter=senders.insert(item,notification_type);

112  //connect it with correct notification type:

114  connect(item, SIGNAL(hasChanged(VarType *)), this, SLOT(changeSlot(VarType *)));
