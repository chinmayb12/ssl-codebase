54 void PluginColorCalibration::mouseEvent( QMouseEvent * event, pixelloc loc) {
55  QTabWidget* tabw = (QTabWidget*) lutw->parentWidget()->parentWidget();
56  if (tabw->currentWidget() == lutw) {
57  if (event->buttons()==Qt::LeftButton) {
58  FrameBuffer * rb=getFrameBuffer();
59  if (rb!=0) {
60  rb->lockRead();
61  int idx=rb->curRead();
62  FrameData * frame = rb->getPointer(idx);
63  if (loc.x < frame->video.getWidth() && loc.y < frame->video.getHeight() && loc.x >=0 && loc.y >=0) {
64  if (frame->video.getWidth() > 1 && frame->video.getHeight() > 1) {
65  yuv color;
66  //if converting entire image then blanking is not needed
67  ColorFormat source_format=frame->video.getColorFormat();
68  if (source_format==COLOR_RGB8) {
69  //plain copy of data
70  rgbImage img(frame->video);
71  color=Conversions::rgb2yuv(img.getPixel(loc.x,loc.y));
72  } else if (source_format==COLOR_YUV444) {
73  yuvImage img(frame->video);
74  color=img.getPixel(loc.x,loc.y);
75  } else if (source_format==COLOR_YUV422_UYVY) {
76  uyvy color2 = *((uyvy*)(frame->video.getData() + (sizeof(uyvy) * (((loc.y * (frame->video.getWidth())) + loc.x) / 2))));
77  color.u=color2.u;
78  color.v=color2.v;
79  if ((loc.x % 2)==0) {
80  color.y=color2.y1;
81  } else {
82  color.y=color2.y2;
83  }
84  } else {
85  //blank it:
86  fprintf(stderr,"Unable to pick color from frame of format: %s\n",Colors::colorFormatToString(source_format).c_str());
87  fprintf(stderr,"Currently supported are rgb8, yuv444, and yuv422 (UYVY).\n");
88  fprintf(stderr,"(Feel free to add more conversions to plugin_colorcalib.cpp).\n");
89  }
90  lutw->samplePixel(color);
91  //img.setPixel(loc.x,loc.y,rgb(255,0,0));
92 
93  if (event->modifiers()!=Qt::NoModifier) {
94  if(event->modifiers() & Qt::ShiftModifier) { //modified 2/1/16, 'control' alone consumed
95  if(event->modifiers() & Qt::ControlModifier) // by QT framework to move a window
96  lutw->add_del_Pixel(color, true, continuing_undo);
97  else
98  lutw->add_del_Pixel(color, false, continuing_undo);
99  }
100  continuing_undo = true;
101  }
102  }
103  }
104  rb->unlockRead();
105  }
106  event->accept();
107  }
108 
109  }
110  else
111  event->ignore();
112 }
