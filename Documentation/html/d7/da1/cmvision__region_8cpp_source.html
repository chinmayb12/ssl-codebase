37 void RegionProcessing::encodeRuns(Image<raw8> * tmap, CMVision::RunList * runlist)
38 // Changes the flat array version of the thresholded image into a run
39 // length encoded version, which speeds up later processing since we
40 // only have to look at the points where values change.
41 {
42 
43  int max_runs = runlist->getMaxRuns();
44  CMVision::Run * runs = runlist->getRunArrayPointer();
45  raw8 * map = tmap->getPixelData();
46  int width=tmap->getWidth();
47  int height=tmap->getHeight();
48 
49 
50  raw8 clear(0);
51  raw8 m;
52  raw8 *row;
53  int x,y,j,l;
54  CMVision::Run r;
55 
56  r.next = 0;
57 
58  j = 0;
59  for(y=0; y<height; y++){
60  row = &map[y * width];
61 
62  r.y = y;
63 
64  x = 0;
65  while(x < width){
66  m = row[x];
67  r.x = x;
68 
69  l = x;
70 
71  //fix by Stefan: stop if x==row-width
72  //(and don't access the row array in that case as it could cause a segfault)
73  //Note that the left argument of the && operator is always evaluated first and as
74  //such this expression should be safe.
75  while(x != width && row[x] == m) x++;
76 
77  if(m != clear || x==width) {
78  r.color = m;
79  r.width = x - l;
80  r.parent = j;
81  runs[j++] = r;
82 
83  if(j >= max_runs){
84  runlist->setUsedRuns(j);
85  return;
86  }
87  }
88  }
89  }
90 
91  runlist->setUsedRuns(j);
92 }

97 void RegionProcessing::connectComponents(CMVision::RunList * runlist)
98 // Connect components using four-connecteness so that the runs each
99 // identify the global parent of the connected region they are a part
100 // of. It does this by scanning adjacent rows and merging where
101 // similar colors overlap. Used to be union by rank w/ path
102 // compression, but now is just uses path compression as the global
103 // parent index is a simpler rank bound in practice.
104 // WARNING: This code is complicated. I'm pretty sure it's a correct
105 // implementation, but minor changes can easily cause big problems.
106 // Read the papers on this library and have a good understanding of
107 // tree-based union find before you touch it
108 {
109 
110  CMVision::Run * map=runlist->getRunArrayPointer();
111  int num = runlist->getUsedRuns();
112  int l1,l2;
113  CMVision::Run r1,r2;
114  int i,j,s;
115 
116  // l2 starts on first scan line, l1 starts on second
117  l2 = 0;
118  l1 = 1;
119  while(map[l1].y == 0) l1++; // skip first line
120 
121  // Do rest in lock step
122  r1 = map[l1];
123  r2 = map[l2];
124  s = l1;
125  while(l1 < num){
126  /*
127  printf("%6d:(%3d,%3d,%3d) %6d:(%3d,%3d,%3d)\n",
128  l1,r1.x,r1.y,r1.width,
129  l2,r2.x,r2.y,r2.width);
130  */
131 
132  if(r1.color==r2.color && r1.color.v!=0) {
133  // case 1: r2.x <= r1.x < r2.x + r2.width
134  // case 2: r1.x <= r2.x < r1.x + r1.width
135  if((r2.x<=r1.x && r1.x<r2.x+r2.width) ||
136  (r1.x<=r2.x && r2.x<r1.x+r1.width)){
137  if(s != l1){
138  // if we didn't have a parent already, just take this one
139  map[l1].parent = r1.parent = r2.parent;
140  s = l1;
141  }else if(r1.parent != r2.parent){
142  // otherwise union two parents if they are different
143 
144  // find terminal roots of each path up tree
145  i = r1.parent;
146  while(i != map[i].parent) i = map[i].parent;
147  j = r2.parent;
148  while(j != map[j].parent) j = map[j].parent;
149 
150  // union and compress paths; use smaller of two possible
151  // representative indicies to preserve DAG property
152  if(i < j){
153  map[j].parent = i;
154  map[l1].parent = map[l2].parent = r1.parent = r2.parent = i;
155  }else{
156  map[i].parent = j;
157  map[l1].parent = map[l2].parent = r1.parent = r2.parent = j;
158  }
159  }
160  }
161  }
162 
163  // Move to next point where values may change
164  i = (r2.x + r2.width) - (r1.x + r1.width);
165  if(i >= 0) r1 = map[++l1];
166  if(i <= 0) r2 = map[++l2];
167  }
168 
169  // Now we need to compress all parent paths
170  for(i=0; i<num; i++){
171  j = map[i].parent;
172  map[i].parent = map[j].parent;
173  }
174 }

178 void RegionProcessing::extractRegions(CMVision::RegionList * reglist, CMVision::RunList * runlist)
179 // Takes the list of runs and formats them into a region table,
180 // gathering the various statistics along the way. num is the number
181 // of runs in the rmap array, and the number of unique regions in
182 // reg[] (bounded by max_reg) is returned. Implemented as a single
183 // pass over the array of runs.
184 {
185  int b,i,n,a;
186  CMVision::Run r;
187  CMVision::Region * reg = reglist->getRegionArrayPointer();
188  CMVision::Run * rmap = runlist->getRunArrayPointer();
189  int max_reg=reglist->getMaxRegions();
190  int num = runlist->getUsedRuns();
191 
192  n = 0;
193 
194  for(i=0; i<num; i++){
195  if(rmap[i].color.v!=0){
196  r = rmap[i];
197  if(r.parent == i){
198  // Add new region if this run is a root (i.e. self parented)
199  rmap[i].parent = b = n; // renumber to point to region id
200  reg[b].color = r.color;
201  reg[b].area = r.width;
202  reg[b].x1 = r.x;
203  reg[b].y1 = r.y;
204  reg[b].x2 = r.x + r.width;
205  reg[b].y2 = r.y;
206  reg[b].cen_x = rangeSum(r.x,r.width);
207  reg[b].cen_y = r.y * r.width;
208  reg[b].run_start = i;
209  reg[b].iterator_id = i; // temporarily use to store last run
210  n++;
211  if(n >= max_reg) {
212  reglist->setUsedRegions(max_reg);
213  return;
214  }
215  }else{
216  // Otherwise update region stats incrementally
217  b = rmap[r.parent].parent;
218  rmap[i].parent = b; // update parent to identify region id
219  reg[b].area += r.width;
220  reg[b].x2 = max(r.x + r.width,reg[b].x2);
221  reg[b].x1 = min((int)r.x,reg[b].x1);
222  reg[b].y2 = r.y; // last set by lowest run
223  reg[b].cen_x += rangeSum(r.x,r.width);
224  reg[b].cen_y += r.y * r.width;
225  // set previous run to point to this one as next
226  rmap[reg[b].iterator_id].next = i;
227  reg[b].iterator_id = i;
228  }
229  }
230  }
231 
232  // calculate centroids from stored sums
233  for(i=0; i<n; i++){
234  a = reg[i].area;
235  reg[i].cen_x = (float)reg[i].cen_x / a;
236  reg[i].cen_y = (float)reg[i].cen_y / a;
237  rmap[reg[i].iterator_id].next = 0; // -1;
238  reg[i].iterator_id = 0;
239  reg[i].x2--; // change to inclusive range
240  }
241 
242  reglist->setUsedRegions(n);
243  return;
244 }

250 int RegionProcessing::separateRegions(CMVision::ColorRegionList * colorlist, CMVision::RegionList * reglist, int min_area)
251 // Splits the various regions in the region table a separate list for
252 // each color. The lists are threaded through the table using the
253 // region's 'next' field. Returns the maximal area of the regions,
254 // which can be used later to speed up sorting.
255 {
256  CMVision::Region * p;
257  int i; // ,l;
258  uint8_t c;
259  int area,max_area;
260  int num_regions=reglist->getUsedRegions();
261  CMVision::Region * reg = reglist->getRegionArrayPointer();
262  int num_colors=colorlist->getNumColorRegions();
263  CMVision::RegionLinkedList * color=colorlist->getColorRegionArrayPointer();
264 
265  // clear out the region list head table
266  for(i=0; i<num_colors; i++){
267  color[i].reset();
268  }
269 
270  // step over the table, adding successive
271  // regions to the front of each list
272  max_area = 0;
273  for(i=0; i<num_regions; i++){
274  p = &reg[i];
275  c = p->color.v;
276  area = p->area;
277  if (c >= num_colors) {
278  printf("Found a color of index %d...but colorlist is only allocated for a max index of %d\n",c,num_colors-1);
279  } else {
280  if(area >= min_area){
281  if(area > max_area) max_area = area;
282  color[c].insertFront(p);
283  }
284  }
285  }
286 
287  return(max_area);
288 }

301 CMVision::Region * RegionProcessing::sortRegionListByArea(CMVision::Region *list,int passes)
302 // Sorts a list of regions by their area field.
303 // Uses a linked list based radix sort to process the list.
304 {
305  CMVision::Region *tbl[CMV_RADIX],*p,*pn;
306  int slot,shift;
307  int i,j;
308 
309  // Handle trivial cases
310  if(!list || !list->next) return(list);
311 
312  // Initialize table
313  for(j=0; j<CMV_RADIX; j++) tbl[j] = NULL;
314 
315  for(i=0; i<passes; i++){
316  // split list into buckets
317  shift = CMV_RBITS * i;
318  p = list;
319  while(p){
320  pn = p->next;
321  slot = ((p->area) >> shift) & CMV_RMASK;
322  p->next = tbl[slot];
323  tbl[slot] = p;
324  p = pn;
325  }
326 
327  // integrate back into partially ordered list
328  list = NULL;
329  for(j=0; j<CMV_RADIX; j++){
330  p = tbl[j];
331  tbl[j] = NULL; // clear out table for next pass
332  while(p){
333  pn = p->next;
334  p->next = list;
335  list = p;
336  p = pn;
337  }
338  }
339  }
340 
341  return(list);
342 }

344 void RegionProcessing::sortRegions(CMVision::ColorRegionList * colors,int max_area)
345 // Sorts entire region table by area, using the above
346 // function to sort each threaded region list.
347 {
348  int i,p;
349  // do minimal number of passes sufficient to touch all set bits
350  p = 0;
351  while(max_area != 0){
352  max_area >>= CMV_RBITS;
353  p++;
354  }
355 
356  int num_colors=colors->getNumColorRegions();
357  CMVision::RegionLinkedList * color = colors->getColorRegionArrayPointer();
358  // sort each list
359  for(i=0; i<num_colors; i++){
360  color[i].setFront(sortRegionListByArea(color[i].getInitialElement(),p));
361  }
362 }
