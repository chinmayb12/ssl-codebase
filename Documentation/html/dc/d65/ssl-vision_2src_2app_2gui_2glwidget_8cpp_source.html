239 void GLWidget::myGLdraw() {
240  glPushAttrib ( GL_ALL_ATTRIB_BITS );
241  glMatrixMode ( GL_PROJECTION );
242  glLoadIdentity();
243  glPushMatrix();
244 
245  glMatrixMode ( GL_MODELVIEW );
246  glLoadIdentity();
247  glPushMatrix();
248 
249  glClear(GL_COLOR_BUFFER_BIT | GL_DEPTH_BUFFER_BIT);
250  if ( rb!=0 ) {
251  rb->lockRead();
252  int idx=rb->curRead();
253  FrameData * frame = rb->getPointer ( idx );
254  VisualizationFrame * vis_frame=(VisualizationFrame *)(frame->map.get("vis_frame"));
255  if (vis_frame!=0 && vis_frame->valid==true && vis_frame->data.getData() != 0 && vis_frame->data.getWidth() >= 1 && vis_frame->data.getHeight() >=1 ) {
256  rgbImage & img = vis_frame->data;
257  if ( img.getWidth() > 1 && img.getHeight() > 1 ) {
258  glPushMatrix();
259  zoom.setup ( img.getWidth(), img.getHeight(), vpW,vpH,true );
260  pixelloc orig=zoom.zoom ( 0, 0 );
261 
262  glPushMatrix();
263  glRasterPos2i ( 0,0 );
264  glBitmap ( 0,0,0,0,orig.x,-orig.y,0 );
265  glPixelZoom ( zoom.getZoom() * zoom.getFlipXval(),zoom.getZoom() * zoom.getFlipYval() * -1.0 );
266  glDrawPixels ( img.getWidth(), img.getHeight(), GL_RGB, GL_UNSIGNED_BYTE, img.getData() );
267  glPopMatrix();
268 
269  glPopMatrix();
270  }
271 
272  }
273  rb->unlockRead();
274  }

276  glMatrixMode ( GL_MODELVIEW );
277  glPopMatrix();
