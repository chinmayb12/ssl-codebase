248  vector<VarType *> VarType::readChildrenHelper(XMLNode & parent , vector<VarType *> existing_children, bool only_update_existing, bool blind_append)
249  {
250  //this again does non-destructive integration
251  //As a result we update any predefined structures and
252  //append any types that don't exist yet.
253  vector<VarType *> result=existing_children;
254  int n=parent.nChildNode("Var");
255  int i,myIterator=0;
256 
257  //iterate through them and check if we already exist
258  set<VarType *> unmatched_children;
259  for (unsigned int j=0;j<existing_children.size();j++) {
260  unmatched_children.insert(existing_children[j]);
261  }
262  for (i=0; i<n; i++) {
263  XMLNode t=parent.getChildNode("Var",&myIterator);
264  string sname=fixString(t.getAttribute("name"));
265  string stype=fixString(t.getAttribute("type"));
266  VarTypeId type=VarTypesInstance::getFactory()->stringToType(stype);
267  if (type!=VARTYPE_ID_UNDEFINED) {
268  bool found=false;
269  if (sname!="" && blind_append==false) {
270  //try to find existing child with this name
271  for (unsigned int j=0;j<existing_children.size();j++) {
272  if (existing_children[j]->getName()==sname) {
273  if (existing_children[j]->getType()==type) {
274  existing_children[j]->readXML(t);
275  } else {
276  fprintf(stderr,"Type mismatch between XML and Data-Tree. Object name: %s. XML type was: %s. Internal type was: %s\n",sname.c_str(),stype.c_str(),existing_children[j]->getTypeName().c_str());
277  }
278  unmatched_children.erase(existing_children[j]);
279  found=true;
280  break;
281  //ok matching name was found...update this node
282  }
283  }
284  }
285  if (found==false) {
286  //printf("unable to find: %s in existing\n",sname.c_str());
287  //the item did not already exist in the children...
288  if (only_update_existing==false) {
289  //create new node and append it:
290  VarType * td= VarTypesInstance::getFactory()->newVarType(type);
291  td->setName(sname);
292  td->readXML(t);
293  result.push_back(td);
294  }
295 
296  }
297  } else {
298  fprintf(stderr,"Found var with no or unknown type in XML... type: %s\n",stype.c_str());
299  }
300  }
301  //---fix for recursing tree that's not defined in xml yet:
302  queue<VarType *> _queue;
303  for (set<VarType *>::iterator iter = unmatched_children.begin(); iter!=unmatched_children.end(); iter++) {
304  if (*iter!=0) _queue.push(*iter);
305  }
306  //recurse all unmatched children to make sure we load all externals:
307  while(_queue.empty()==false) {
308  VarType * d = _queue.front();
309  _queue.pop();
310  d->loadExternal();
311  vector<VarType *> children = d->getChildren();
312  int s=children.size();
313  for (int i=0;i<s;i++) {
314  if (children[i]!=0) _queue.push(children[i]);
315  }
316  }
317  //---end of fix
318  return result;
319  }
