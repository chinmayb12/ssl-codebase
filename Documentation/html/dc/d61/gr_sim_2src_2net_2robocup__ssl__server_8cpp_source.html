69  bool success = packet.SerializeToArray(datagram.data(), datagram.size());
70  if(!success) {
71  //TODO: print useful info

73  return false;
74  }
75 
76  mutex.lock();

78  mutex.unlock();
79  if (bytes_sent != datagram.size()) {
80  logStatus(QString("Sending UDP datagram failed (maybe too large?). Size was: %1 byte(s).").arg(datagram.size()), QColor("red"));
81  return false;
