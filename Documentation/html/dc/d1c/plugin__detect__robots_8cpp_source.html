69 void PluginDetectRobots::buildRegionTree(CMVision::ColorRegionList * colorlist) {
70  reg_tree.clear();
71  int num_colors=colorlist->getNumColorRegions();
72  for(int c=0;c<num_colors;c++) {
73  //ONLY ADD ROBOT MARKER COLORS:
74  if (c!= color_id_clear && c!=color_id_field && c!= color_id_ball && c!= color_id_black) {
75  CMVision::Region *reg = colorlist->getRegionList(c).getInitialElement();
76  while(reg!=0) {
77  reg_tree.add(reg);
78  reg = reg->next;
79  }
80  }
81  }
82  reg_tree.build();
83 }

85 ProcessResult PluginDetectRobots::process(FrameData * data, RenderOptions * options)
86 {
87  (void)options;
88  if (data==0) return ProcessingFailed;
89 
90  SSL_DetectionFrame * detection_frame = 0;
91 
92  detection_frame=(SSL_DetectionFrame *)data->map.get("ssl_detection_frame");
93  if (detection_frame == 0) detection_frame=(SSL_DetectionFrame *)data->map.insert("ssl_detection_frame",new SSL_DetectionFrame());
94 
95  //acquire orange region list from data-map:
96  CMVision::ColorRegionList * colorlist;
97  colorlist=(CMVision::ColorRegionList *)data->map.get("cmv_colorlist");
98  if (colorlist==0) {
99  printf("error in robot detection plugin: no region-lists were found!\n");
100  return ProcessingFailed;
101  }
102 
103  //acquire color-labeled image from data-map:
104  const Image<raw8> * image = (Image<raw8> *)(data->map.get("cmv_threshold"));
105  if (image==0) {
106  printf("error in robot detection plugin: no color-thresholded image was found!\n");
107  return ProcessingFailed;
108  }
109 
110  CMPattern::Team * team=0;
111  ::google::protobuf::RepeatedPtrField< ::SSL_DetectionRobot >* robotlist=0;
112 
113  int color_id;
114  int num_robots;
115  CMPattern::TeamDetector * detector;
116  //TODO: lookup color label from LUT
117 
118  buildRegionTree(colorlist);
119  bool need_reinit=_notifier.hasChanged();
120 
121  for (int team_i = 0; team_i < 2; team_i++) {
122  //team_i: 0==blue, 1==yellow
123  if (team_i==0) {
124  color_id=color_id_blue;
125  team=global_team_selector_blue->getSelectedTeam();
126  num_robots=global_team_selector_blue->getNumberRobots();
127  detection_frame->clear_robots_blue();
128  robotlist=detection_frame->mutable_robots_blue();
129  detector=team_detector_blue;
130  } else {
131  color_id=color_id_yellow;
132  team=global_team_selector_yellow->getSelectedTeam();
133  num_robots=global_team_selector_yellow->getNumberRobots();
134  detection_frame->clear_robots_yellow();
135  robotlist=detection_frame->mutable_robots_yellow();
136  detector=team_detector_yellow;
137  }
138  if (team!=0) {
139  if (need_reinit) {
140  detector->init(team);
141  }
142 
143  detector->update(robotlist, color_id, num_robots, image, colorlist, reg_tree);
144  } else {
145  _notifier.changeSlotOtherChange();
146  }
147 
148 // printf("DETECTED %d robots on team %d\n",robotlist->size(),team_i);
149 // fflush(stdout);
150  }
151  return ProcessingOk;
152 
153 }
