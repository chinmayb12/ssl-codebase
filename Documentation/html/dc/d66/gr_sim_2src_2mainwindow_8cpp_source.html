302 void MainWindow::update()
303 {
304  if (glwidget->ssl->g->isGraphicsEnabled()) glwidget->updateGL();
305  else glwidget->step();
306 
307  int R = robotIndex(glwidget->Current_robot,glwidget->Current_team);
308 
309  const dReal* vv = dBodyGetLinearVel(glwidget->ssl->robots[R]->chassis->body);
310  static dVector3 lvv;
311  dVector3 aa;
312  aa[0]=(vv[0]-lvv[0])/configwidget->DeltaTime();
313  aa[1]=(vv[1]-lvv[1])/configwidget->DeltaTime();
314  aa[2]=(vv[2]-lvv[2])/configwidget->DeltaTime();
315  robotwidget->vellabel->setText(QString::number(sqrt(vv[0]*vv[0]+vv[1]*vv[1]+vv[2]*vv[2]),'f',3));
316  robotwidget->acclabel->setText(QString::number(sqrt(aa[0]*aa[0]+aa[1]*aa[1]+aa[2]*aa[2]),'f',3));
317  lvv[0]=vv[0];
318  lvv[1]=vv[1];
319  lvv[2]=vv[2];
320  QString ss;
321  fpslabel->setText(QString("Frame rate: %1 fps").arg(ss.sprintf("%06.2f",glwidget->getFPS())));
322  if (glwidget->ssl->selected!=-1)
323  {
324  selectinglabel->setVisible(true);
325  if (glwidget->ssl->selected==-2)
326  {
327  selectinglabel->setText("Ball");
328  }
329  else
330  {
331  int R = glwidget->ssl->selected%ROBOT_COUNT;
332  int T = glwidget->ssl->selected/ROBOT_COUNT;
333  if (T==0) selectinglabel->setText(QString("%1:Blue").arg(R));
334  else selectinglabel->setText(QString("%1:Yellow").arg(R));
335  }
336  }
337  else selectinglabel->setVisible(false);
338  vanishlabel->setVisible(configwidget->vanishing());
339  noiselabel->setVisible(configwidget->noise());
340  cursorlabel->setText(QString("Cursor: [X=%1;Y=%2;Z=%3]").arg(dRealToStr(glwidget->ssl->cursor_x)).arg(dRealToStr(glwidget->ssl->cursor_y)).arg(dRealToStr(glwidget->ssl->cursor_z)));
341  statusWidget->update();
342 }

391 void MainWindow::ballMenuTriggered(QAction* act)
392 {
393  dReal l = configwidget->Field_Length()/2.0;
394  dReal w = configwidget->Field_Width()/2.0;
395  dReal p = l - configwidget->Field_Penalty_Point();
396  if (act->text()==tr("Put on Center")) glwidget->putBall(0,0);
397  else if (act->text()==tr("Put on Corner 1")) glwidget->putBall(-l,-w);
398  else if (act->text()==tr("Put on Corner 2")) glwidget->putBall(-l, w);
399  else if (act->text()==tr("Put on Corner 3")) glwidget->putBall( l,-w);
400  else if (act->text()==tr("Put on Corner 4")) glwidget->putBall( l, w);
401  else if (act->text()==tr("Put on Penalty 1")) glwidget->putBall( p, 0);
402  else if (act->text()==tr("Put on Penalty 2")) glwidget->putBall(-p, 0);
