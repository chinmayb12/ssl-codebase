72 class TeamSelector : public QObject {
73 Q_OBJECT
74 signals:
75  void signalTeamDataChanged();
76 protected:
77  TeamDetectorSettings * _detector_settings;
78  VarList * _settings;
79  VarStringEnum * _selector;
80  VarInt * _num_robots;
81  Team * current_team;
82  void update() {
83  vector<Team *> teams = _detector_settings->getTeams();
84  _selector->setSize(teams.size(), "");
85  int old_select_index = _selector->getIndex();
86  for (unsigned int i=0;i<teams.size();i++) {
87  _selector->setLabel(i,teams[i]-> _team_name->getString());
88  }
89  Team * new_team=0;
90  if (teams.size() > 0) {
91  if (old_select_index >= 0 && old_select_index < (int)teams.size()) {
92  _selector->selectIndex(old_select_index);
93  } else {
94  _selector->selectIndex(0);
95  }
96  new_team=_detector_settings->getTeam(_selector->getIndex());
97  } else {
98  new_team=0;
99  }
100  if (new_team!=current_team) {
101  if (current_team!=0) {
102  disconnect(current_team,SIGNAL(signalChangeOccured(VarType*)),this,SLOT(slotTeamDataChanged()));
103  }
104  if (new_team!=0) {
105  connect(new_team,SIGNAL(signalChangeOccured(VarType*)),this,SLOT(slotTeamDataChanged()));
106  }
107  current_team=new_team;
108  }
109  emit(signalTeamDataChanged());
110  }
111 protected slots:
112  void slotTeamInfoChanged() {
113  update();
114  }
115  void slotTeamDataChanged() {
116  emit(signalTeamDataChanged());
117  }
118 public:
119  TeamSelector(string label, TeamDetectorSettings * detector_settings) {
120  _detector_settings=detector_settings;
121  connect(_detector_settings,SIGNAL(teamInfoChanged()),this,SLOT(slotTeamInfoChanged()));
122  current_team=0;
123  _settings= new VarList(label);
124  _settings->addChild(_selector = new VarStringEnum("Team",""));
125  _selector->addFlags(VARTYPE_FLAG_NOLOAD_ENUM_CHILDREN);
126  _settings->addChild(_num_robots = new VarInt("Max Robots",6));
127  update();
128  }
129  VarList * getSettings() {
130  return _settings;
131  }
132  Team * getSelectedTeam() {
133  if (_selector->getIndex() == -1) {
134  update();
135  } else if (current_team!=_detector_settings->getTeam(_selector->getIndex())) {
136  update();
137  }
138  return current_team;
139  }
140  int getNumberRobots() {
141  return _num_robots->getInt();
142  }
143  ~TeamSelector() {
144  delete _settings;
145  }
146 };
