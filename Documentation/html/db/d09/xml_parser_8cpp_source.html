606  // sequences &amp;, &quot;, &apos;, &lt;, &gt; and replace them by the characters
607  // &,",',<,>. This function is used internally by the XML Parser. All the calls to
608  // the XML library will always gives you back "decoded" strings.

749  // pcbToken contains the number of characters that have been read.
750  static NextToken GetNextToken(XML *pXML, int *pcbToken, enum XMLTokenTypeTag *pType)
751  {

756  result.pClr=NULL; // prevent warning
757 
758  // Find next non-white space character
759  do { indexStart=pXML->nIndex; ch=getNextChar(pXML); } while XML_isSPACECHAR(ch);
760 
761  if (ch)

763  // Cache the current string pointer
764  result.pStr = &pXML->lpXML[indexStart];

767  // does not need formatting).
768  ALLXMLClearTag *ctag=XMLClearTags;

772  {

774  pXML->nIndex+=ctag->openTagLen-1;
775  *pType=eTokenClear;

777  }
778  ctag++;

782  switch(ch)
783  {
784  // Check for quotes
785  case _T('\''):
786  case _T('\"'):
787  // Type of token
788  *pType = eTokenQuotedText;
789  chTemp = ch;
790 
791  // Set the size
792  nFoundMatch = FALSE;
793 
794  // Search through the string to find a matching quote

796  {
797  if (ch==chTemp) { nFoundMatch = TRUE; break; }
798  if (ch==_T('<')) break;
799  }
800 
801  // If we failed to find a matching quote
802  if (nFoundMatch == FALSE)
803  {
804  pXML->nIndex=indexStart+1;
805  nIsText=TRUE;
806  break;
807  }
808 
809  // 4.02.2002
810  // if (FindNonWhiteSpace(pXML)) pXML->nIndex--;
811 
812  break;

814  // Equals (used with attribute values)
815  case _T('='):
816  *pType = eTokenEquals;
817  break;
818 
819  // Close tag
820  case _T('>'):
821  *pType = eTokenCloseTag;
822  break;
823 
824  // Check for tag start and tag end
825  case _T('<'):
826 
827  // Peek at the next character to see if we have an end tag '</',
828  // or an xml declaration '<?'
829  chTemp = pXML->lpXML[pXML->nIndex];
830 
831  // If we have a tag end...
832  if (chTemp == _T('/'))
833  {
834  // Set the type and ensure we point at the next character
835  getNextChar(pXML);
836  *pType = eTokenTagEnd;
837  }
838 
839  // If we have an XML declaration tag
840  else if (chTemp == _T('?'))
841  {
842 
843  // Set the type and ensure we point at the next character
844  getNextChar(pXML);
845  *pType = eTokenDeclaration;
846  }
847 
848  // Otherwise we must have a start tag
849  else
850  {
851  *pType = eTokenTagStart;
852  }
853  break;
854 
855  // Check to see if we have a short hand type end tag ('/>').
856  case _T('/'):
857 
858  // Peek at the next character to see if we have a short end tag '/>'
859  chTemp = pXML->lpXML[pXML->nIndex];
860 
861  // If we have a short hand end tag...
862  if (chTemp == _T('>'))
863  {
864  // Set the type and ensure we point at the next character
865  getNextChar(pXML);
866  *pType = eTokenShortHandClose;
867  break;
868  }
869 
870  // If we haven't found a short hand closing tag then drop into the
871  // text process
872 
873  // Other characters
874  default:
875  nIsText = TRUE;
876  }
877 
878  // If this is a TEXT node
879  if (nIsText)
880  {
881  // Indicate we are dealing with text
882  *pType = eTokenText;
883  while((ch = getNextChar(pXML)))
884  {
885  if XML_isSPACECHAR(ch)
886  {
887  indexStart++; break;
888 
889  } else if (ch==_T('/'))
890  {
891  // If we find a slash then this maybe text or a short hand end tag
892  // Peek at the next character to see it we have short hand end tag
893  ch=pXML->lpXML[pXML->nIndex];
894  // If we found a short hand end tag then we need to exit the loop
895  if (ch==_T('>')) { pXML->nIndex--; break; }
896 
897  } else if ((ch==_T('<'))||(ch==_T('>'))||(ch==_T('=')))
898  {
899  pXML->nIndex--; break;
900  }
901  }
902  }
903  *pcbToken = pXML->nIndex-indexStart;
904  } else
905  {
906  // If we failed to obtain a valid character
907  *pcbToken = 0;
908  *pType = eTokenError;
909  result.pStr=NULL;
910  }
911 
912  return result;

1301  if (type != eTokenText)

1304  return FALSE;
1305  }

1507  case eTokenTagStart: /* 'Attr = <' */
