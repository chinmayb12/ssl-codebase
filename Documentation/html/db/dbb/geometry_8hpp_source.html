195 // Detects if a line segment intersects a circle or not
196 template <class T>
197 inline bool intersects(const Point2D<T>& point1,
198  const Point2D<T>& point2,
199  const Point2D<T>& center,
200  T radius)
201 {
202  /* Source of algorithm used: http://stackoverflow.com/questions/1073336/circle-line-collision-detection */
203 
204  Vector2D<int> d = point2 - point1;
205  Vector2D<int> f = point1 - center;
206 
207  float a = d.dot(d);
208  float b = 2 * f.dot(d);
209  float c = f.dot(f) - radius * radius;
210  float dis = b * b - 4 * a * c;
211 
212  if (dis >= 0)
213  {
214  dis = sqrt(dis);
215  float t1 = (-b + dis) / (2 * a);
216  float t2 = (-b - dis) / (2 * a);
217  if (t1 >= 0 && t1 <= 1)
218  return true;
219  if (t2 >= 0 && t2 <= 1)
220  return true;
221  }
222  return false;
223 }
