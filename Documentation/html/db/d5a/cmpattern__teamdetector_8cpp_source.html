266 double TeamDetector::getRegionArea(const CMVision::Region * reg, double z) const {
267  // calculate area of bounding box in sq mm
268  vector3d a,b;
269  vector2d right(reg->x2+1,reg->y2+1);
270  vector2d left(reg->x1,reg->y1);
271  _camera_params.image2field(a,right,z);
272  _camera_params.image2field(b,left,z);
273  vector3d box = a-b;
274 
275  double box_area = fabs(box.x) * fabs(box.y);
276  int box_pixels = (reg->x2+1 - reg->x1) * (reg->y2+1 - reg->y1);
277 
278  // estimate world coordinate area of region
279  double area = ((double)reg->area) * box_area / ((double)box_pixels);
280 
281  return(area);
282 }
