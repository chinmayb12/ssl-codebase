123 double CameraParameters::radialDistortion(double ru) const {
124  if ((distortion->getDouble())<=DBL_MIN)
125  return ru;
126  double rd = 0;
127  double a = distortion->getDouble();

296  std::vector<GVector::vector2d<double> >::iterator it_p_i = p_i.begin();
297

301  field2image(*it_p_f, proj_p, p);
302  chisqr += (proj_p.x - it_p_i->x) * (proj_p.x - it_p_i->x) * cov_cx_inv +
303  (proj_p.y - it_p_i->y) * (proj_p.y - it_p_i->y) * cov_cy_inv;
304  }
305

307  if (cal_type & FULL_ESTIMATION)

309  std::vector<CalibrationData>::iterator ls_it = calibrationSegments.begin();
310 
311  int i = 0;
312  for (; ls_it != calibrationSegments.end(); ls_it++) {
313  std::vector< std::pair<GVector::vector2d<double>,bool> >::iterator
314  imgPts_it = (*ls_it).imgPts.begin();
315  for (; imgPts_it != (*ls_it).imgPts.end(); imgPts_it++) {
316  // Integrate only if a valid point on line
317  if (imgPts_it->second) {
318  GVector::vector2d<double> proj_p;
319  double alpha = p_alpha(i) + p(STATE_SPACE_DIMENSION + i);
320 
321  // Calculate point on segment
322  GVector::vector3d<double> alpha_point;
323  if (ls_it->straightLine) {
324  alpha_point = alpha * (*ls_it).p1 + (1.0 - alpha) * (*ls_it).p2;
325  } else {
326  double theta = alpha * (*ls_it).theta1 + (1.0 - alpha) * (*ls_it).theta2;
327  alpha_point = ls_it->center +
328  ls_it->radius*GVector::vector3d<double>(
329  cos(theta),sin(theta),0.0);
330  }
331 
332  // Project into image plane
333  field2image(alpha_point, proj_p, p);
334 
335  chisqr += (proj_p.x-imgPts_it->first.x) *
336  (proj_p.x-imgPts_it->first.x) * cov_lsx_inv +
337  (proj_p.y - imgPts_it->first.y) *
338  (proj_p.y - imgPts_it->first.y) * cov_lsy_inv;
339  i++;
340  }
341  }
342  }
343  }

542  } else {
543  double theta = my_alpha * (*ls_it).theta1 + (1.0 - my_alpha) *
544  (*ls_it).theta2;

546  ls_it->radius*GVector::vector3d<double>(cos(theta),
547  sin(theta),0.0);

552  field2image(alpha_point, proj_p_diff);
553  J(0,STATE_SPACE_DIMENSION + i) =
554  ((proj_p_diff.x - (*pts_it).first.x) - proj_p.x) / epsilon;
555  J(1,STATE_SPACE_DIMENSION + i) =
556  ((proj_p_diff.y - (*pts_it).first.y) - proj_p.y) / epsilon;
557 
558  alpha += J.transpose() * cov_ls_inv * J;
559  beta += J.transpose() * cov_ls_inv *
560  Eigen::Vector2d(proj_p.x, proj_p.y);
561  i++;
562  }
563  }
564  }
565  }

567  // Augment alpha
568  alpha += Eigen::MatrixXd::Identity(
569  STATE_SPACE_DIMENSION + num_alpha, STATE_SPACE_DIMENSION + num_alpha)
570  * lambda;
571 
572  // Solve for x
573  Eigen::VectorXd new_p(STATE_SPACE_DIMENSION + num_alpha);
574 
575  // Due to an API change we need to check for
576  // the right call at compile time
577 #ifdef EIGEN_WORLD_VERSION
578  // alpha.llt().solve(-beta, &new_p); -- modify 1/15/16
579  // -- move to Eigen3 structure -
580  // -- http://eigen.tuxfamily.org/dox/Eigen2ToEigen3.html
581  new_p = alpha.llt().solve(-beta);
582 #else
583  Eigen::Cholesky<Eigen::MatrixXd> c(alpha);
584  new_p = c.solve(-beta);
585 #endif

588  double chisqr = calc_chisqr(p_f, p_i, new_p, cal_type);

597  tx->setDouble(tx->getDouble() + new_p[T_1]);
598  ty->setDouble(ty->getDouble() + new_p[T_2]);
599  tz->setDouble(tz->getDouble() + new_p[T_3]);
600

602  GVector::vector3d<double> aa_diff(new_p[Q_1], new_p[Q_2], new_p[Q_3]);
603  q_diff.setAxis(aa_diff.norm(), aa_diff.length());

605  q0->getDouble(),q1->getDouble(),q2->getDouble(),q3->getDouble());
606  q_field2cam.norm();

608  q0->setDouble(q_field2cam.x);
609  q1->setDouble(q_field2cam.y);
610  q2->setDouble(q_field2cam.z);
611  q3->setDouble(q_field2cam.w);
612

614  p_alpha[i] += new_p[STATE_SPACE_DIMENSION + i];

617  // of both possible projections

629  } else {

631  convergence_counter = 0;
632  }

637 #endif
