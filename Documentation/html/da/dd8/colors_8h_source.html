477 class ColorConversions {
478 public:
479 //input/output ranges:
480 //h: 0-360
481 //s,v 0-1
482 //rgb: 8-bit ints
483  static void hsv2rgb(double h, double s, double v, rgb & v_out) {
484 
485 int j;
486 double rd, gd, bd;
487 double f, p, q, t;
488 
489 if(h==-1 || s==0.0)
490 {
491  rd = v;
492  gd = v;
493  bd = v;
494 }
495 else
496 {
497  if(h==360.0) h = 0.0;
498  h = h / 60.0;
499  j = (int) floor(h);
500  if(j<0) j=0;
501  f = h - j;
502  p = v * (1-s);
503  q = v * (1 - (s*f));
504  t = v * (1 - (s*(1 - f)));
505 
506  switch(j)
507  {
508  case 0: rd = v; gd = t; bd = p; break;
509  case 1: rd = q; gd = v; bd = p; break;
510  case 2: rd = p; gd = v; bd = t; break;
511  case 3: rd = p; gd = q; bd = v; break;
512  case 4: rd = t; gd = p; bd = v; break;
513  case 5: rd = v; gd = p; bd = q; break;
514  default:rd = v; gd = t; bd = p; break;
515  }
516 }
517 
518 v_out.r = (int) floor((rd * 255.0) + 0.5);
519 v_out.g = (int) floor((gd * 255.0) + 0.5);
520 v_out.b = (int) floor((bd * 255.0) + 0.5);
521  }
522 };
