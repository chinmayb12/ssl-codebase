142 QPainterPath GraphScene::makePath(TimeLine * l, int minidx, int maxidx, bool zero_edges) {
143  QPainterPath path;
144  QPointF p1;
145  QPointF p2;
146  for (int j=minidx;j<=maxidx;j++) {
147  p2.setX(((TimeIndex*)l->at(j))->getTime());
148  p2.setY(l->at(j)->getData()->getValue());
149  if (zero_edges==true && j==minidx) {
150  path.moveTo(QPointF(p2.x(),0));
151  }
152  if (draw_smooth==false) {
153  if (j==minidx && zero_edges ==false) {
154  path.moveTo(p2);
155  } else {
156  path.lineTo(p2);
157  }
158  } else {
159  if (j==minidx) {
160  if (zero_edges) {
161  path.lineTo(p2);
162  } else {
163  path.moveTo(p2);
164  }
165  } else {
166  double distance = p2.x() - p1.x();
167  path.cubicTo(p1.x() + distance / 2, p1.y(),
168  p1.x() + distance / 2, p2.y(),
169  p2.x(), p2.y());
170  }
171  }
172  if (zero_edges==true && j==maxidx) {
173  path.lineTo(QPointF(p2.x(),0));
174  }
175 
176  p1=p2;
177  }
178 
179  return path;
180 }
