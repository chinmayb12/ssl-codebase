119 class GLSmoothCamera : public GLCamera {
120  //this camera smoothly scrolls to a different camera target using a basic mass-spring system
121 private:
122  vector3d v_forward;
123  vector3d v_pos;
124  double v_angle;
125  bool prevent_overshoot;
126  double k_acc;
127  double k_damp;
128 public:
129  GLCamera target;
130  GLSmoothCamera(bool _prevent_overshoot=true, double k_acceleration=20.0, double k_damping=10.0) {
131  k_acc=k_acceleration;
132  k_damp=k_damping;
133  v_forward.set(0.0,0.0,0.0);
134  v_pos.set(0.0,0.0,0.0);
135  v_angle = 0.0;
136  prevent_overshoot=_prevent_overshoot;
137  };
138  virtual ~GLSmoothCamera() {};
139  void step(const double dt) {
140  //compute transition "delta" vectors:
141  //this is the actual distance to the target
142  vector3d d_forward = target.forward - forward;
143  vector3d d_pos = target.pos - pos;
144  double d_angle = GVector::shortest_angle(rot.getZvector(),target.rot.getZvector());
145  if (std::isnan(d_angle)) d_angle=0.0;
146  //compute spring forces:
147 
148  vector3d f_forward = d_forward * k_acc - v_forward * k_damp;
149  vector3d f_pos = d_pos * k_acc - v_pos * k_damp;
150  double f_angle = d_angle * k_acc - v_angle * k_damp;
151 
152  //integrate:
153  vector3d vnew_forward = v_forward + (f_forward * dt);
154  vector3d vnew_pos = v_pos + (f_pos * dt);
155  double vnew_angle = v_angle + (f_angle * dt);
156 
157  //compute how far we will move (our 'applied' delta positions)
158  vector3d delta_forward = (vnew_forward+v_forward) * 0.5 * dt;
159  vector3d delta_pos = (vnew_pos+v_pos) * 0.5 * dt;
160  double delta_angle = (vnew_angle+v_angle) * 0.5 * dt;
161 
162  if (prevent_overshoot) {
163  //hardcoded fix against overshooting: if applied deltas are bigger than target deltas, then shrink em
164  //we will now never, ever overshoot...although it might be desirable if the target is moving
165  if (delta_forward.length() > d_forward.length()) delta_forward=d_forward;
166  if (delta_pos.length() > d_pos.length()) delta_pos=d_pos;
167  if (delta_angle > d_angle) delta_angle=d_angle;
168  }
169 
170  //add deltas to position.
171  pos+=delta_pos;
172  forward+=delta_forward;
173 
174  //perform angular blending:
175  double blend_n=delta_angle / d_angle;
176 
177  if (blend_n==0.0 || d_angle==0.0 || delta_angle==0.0 || std::isnan(blend_n)) {
178  //don't rotate!
179  } else {
180  rot.blend(blend_n,target.rot);
181  }
182  fflush(stdout);
183  v_forward=vnew_forward;
184  v_pos=vnew_pos;
185  v_angle=vnew_angle;
186 
187  };
188 
189 };
