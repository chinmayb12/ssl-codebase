85 bool CaptureGenerator::copyAndConvertFrame ( const RawImage & src, RawImage & target )
86 {
87 #ifndef VDATA_NO_QT
88  mutex.lock();
89 #endif
90  ColorFormat output_fmt = Colors::stringToColorFormat ( v_colorout->getSelection().c_str() );
91  ColorFormat src_fmt=src.getColorFormat();
92 
93  if ( target.getData() ==0 ) {
94  target.allocate ( output_fmt, src.getWidth(), src.getHeight() );
95  } else {
96  target.ensure_allocation ( output_fmt, src.getWidth(), src.getHeight() );
97  }
98  target.setTime ( src.getTime() );
99 
100  if ( output_fmt == src_fmt ) {
101  if ( src.getData() != 0 ) memcpy ( target.getData(),src.getData(),src.getNumBytes() );
102  } else if ( src_fmt == COLOR_RGB8 && output_fmt == COLOR_YUV422_UYVY ) {
103  if ( src.getData() != 0 ) {
104  dc1394_convert_to_YUV422 ( src.getData(), target.getData(), src.getWidth(), src.getHeight(),
105  DC1394_BYTE_ORDER_UYVY, DC1394_COLOR_CODING_RGB8, 8 );
106  }
107  } else {
108  fprintf ( stderr,"Cannot copy and convert frame...unknown conversion selected from: %s to %s\n",
109  Colors::colorFormatToString ( src_fmt ).c_str(),
110  Colors::colorFormatToString ( output_fmt ).c_str() );
111 #ifndef VDATA_NO_QT
112  mutex.unlock();
113 #endif
114  return false;
115  }
116 #ifndef VDATA_NO_QT
117  mutex.unlock();
118 #endif

122 RawImage CaptureGenerator::getFrame()
123 {
124 #ifndef VDATA_NO_QT
125  mutex.lock();
126 #endif
127  limit.waitForNextFrame();
128  result.setColorFormat ( COLOR_RGB8 );
129  result.setTime ( GetTimeSec() );
130  result.allocate ( COLOR_RGB8,v_width->getInt(),v_height->getInt() );
131  rgbImage img;
132  img.fromRawImage(result);
133 
134  if (v_test_image->getBool()) {
135  int w = result.getWidth();
136  int h = result.getHeight();
137  int n_colors = 8;
138  int slice_width = w/n_colors;
139  rgb color=RGB::Black;
140  rgb color2;
141  for (int x = 0 ; x < w; x++) {
142  int c_idx= x/slice_width;
143  if (c_idx==0) {
144  color=RGB::White;
145  } else if (c_idx==1) {
146  color=RGB::Red;
147  } else if (c_idx==2) {
148  color=RGB::Green;
149  } else if (c_idx==3) {
150  color=RGB::Blue;
151  } else if (c_idx==4) {
152  color=RGB::Cyan;
153  } else if (c_idx==5) {
154  color=RGB::Pink;
155  } else if (c_idx==6) {
156  color=RGB::Yellow;
157  } else if (c_idx==7) {
158  color=RGB::Black;
159  }
160  for (int y = 0 ; y < h; y++) {
161  color2.r= max(0,min(255,((int)color.r * (h-1-y))/h));
162  color2.g= max(0,min(255,((int)color.g * (h-1-y))/h));
163  color2.b= max(0,min(255,((int)color.b * (h-1-y))/h));
164  img.setPixel(x,y,color2);
165  }
166  }
167  } else {
168  img.fillBlack();
169  }
170 
171 
172 #ifndef VDATA_NO_QT
173  mutex.unlock();
174 #endif
175  return result;
176 }
