48 void VisionStack::process(FrameData * data) {
49  double a=0.0;
50  double b=0.0;
51  unsigned int n=stack.size();
52  VisionPlugin * p;
53  double total=0.0;
54  bool show_timing = false;
55  if (show_timing) printf("----------\n");
56  for (unsigned int i=0;i<n;i++) {
57  p=stack[i];
58  p->lock();
59  a=GetTimeSec();
60  p->process(data,opts);
61  b=GetTimeSec();
62  p->setTimeProcessing(b-a);
63  total+=(p->getTimeProcessing());
64  if (show_timing) {
65  printf("Plugin %s: %fms\n",p->getName().c_str(), p->getTimeProcessing() * 1000.0);
66  }
67  p->unlock();
68  }
69  if (show_timing) printf("Total time: %fms\n",total * 1000.0);
70  //counter_proc+=1.0;
71 }
