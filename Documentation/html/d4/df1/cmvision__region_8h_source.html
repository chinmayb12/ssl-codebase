223  int w = reg.x2 - reg.x1 + 1;

229  void init(const CMVision::Region *region_list) {
230  reg = region_list;
231 
232  // skip too-large regions in sorted region list
233  while(reg!=0 && reg->area>area.max) reg = reg->next;
234  }
235 
236  const CMVision::Region * getNext()
237  {
238  // terminate when no regions, or no suitably large ones
239  if(reg==0) return(0);
240 
241  // find the next region matching our ranges
242  while(reg!=0) {
243  w = reg->width();
244  h = reg->height();
245 
246  if(reg->area < area.min) return(0);
247  if(width.inside(w) && height.inside(h)){
248  const CMVision::Region *match = reg;
249  reg = reg->next;
250  return(match);
251  }
252 
253  reg = reg->next;
254  }
255  return(0);
256  }

271 protected:
272  //==== Utility Functions ===========================================//
273  // sum of integers over range [x,x+w)
274  inline static int rangeSum(int x,int w)
275  {
276  return(w*(2*x + w-1) / 2);
277  }
278 
279  // sum of integer squares over range [x,x+w)
280  // S(n) = n*(n+1)*(2*n+1) / 6
281  // R(x,w) = S(x+w-1) - S(x-1)
282  // ref: http://mathworld.wolfram.com/SquarePyramidalNumber.html
283  // note: if x+w > 1024, you must use 64-bit ints for correct results
284  inline static int rangeSumSq(int x,int w)
285  {
286  int y = x + w;
287  int rs = y*(y-1)*(2*y-1) - x*(x-1)*(2*x-1);
288  return(rs / 6);
289  }
