197 bool MultiPatternModel::loadSinglePatternImage(const yuvImage & image, YUVLUT * _lut,int idx, float default_object_height) {
198 
199  if (_lut==0) return false;
200  int color_team_id=_lut->getChannelID("Blue");
201  if (color_team_id==-1) {
202  fprintf(stderr,"Unable to find color label 'Blue' in LUT!\n");
203  fflush(stderr);
204  return false;
205  }
206 
207  int color_height_id=_lut->getChannelID("Yellow");
208  if (color_height_id==-1) {
209  fprintf(stderr,"Warning: Unable to find color label 'Yellow' in LUT!\n");
210  fflush(stderr);
211  }
212 
213  vector<int> marker_color_ids;
214  vector<string> marker_color_labels;
215  marker_color_labels.push_back("Pink");
216  marker_color_labels.push_back("White");
217  marker_color_labels.push_back("Cyan");
218  marker_color_labels.push_back("Green");
219 
220  for (unsigned int i = 0 ; i < marker_color_labels.size(); i++) {
221  int id=_lut->getChannelID(marker_color_labels[i]);
222  if (id==-1) {
223  fprintf(stderr,"Unable to find color label '%s' in LUT!\n",marker_color_labels[i].c_str());
224  fflush(stderr);
225  } else {
226  marker_color_ids.push_back(id);
227  }
228  }
229 
230  CMVision::ImageProcessor proc(_lut);
231 
232 /* // remap all dark greens to vision's field green color
233  int n = image.Width() * image.getHeight();
234  rgb green = vision.color[ColorFieldGreen].color;
235  for(int i=0; i<n; i++){
236  rgb c = img.img[i];
237  if(c.red==0 && c.blue==0 && c.green>0 && c.green<=128){
238  img.img[i] = green;
239  }
240  }*/
241 
242  // run low level vision on the image
243 
244  proc.processYUV444(&image,10);
245 
246  CMVision::ColorRegionList * colors = proc.getColorRegionList();
247 
248  const CMVision::Region * reg = colors->getRegionList(color_team_id).getInitialElement();
249 
250  // find center dot
251  if (reg==0) {
252  printf("Info: Ignoring Pattern Image for Pattern ID %d. It is missing the center marker.\n",idx);
253  return false;
254  }
255 
256  vector2f cen(-reg->cen_y,-reg->cen_x);
257 
258  // find height
259  float height = default_object_height;
260  if (color_height_id!=-1) {
261  reg = colors->getRegionList(color_height_id).getInitialElement();
262 
263  if(reg!=0 ){
264  if(reg->width()<1 || reg->width() > 6) {
265  printf("WARNING: No Object Height Indicator Found in Image (for robot id=%d)!\n",idx);
266  } else if ( reg->next != 0) {
267  printf("WARNING: Multiple Height Indicators Found in Image (for robot id=%d)!\n",idx);
268  } else {
269  height = reg->height();
270  }
271  } else {
272  printf("WARNING: No Object Height Indicator Found in Image (for robot id=%d)!\n",idx);
273  }
274  }
275 
276  std::vector<Marker> markers;
277  Marker m;
278  m.reset();
279 
280  for(unsigned int c=0; c<marker_color_ids.size(); c++) {
281  reg = colors->getRegionList(marker_color_ids[c]).getInitialElement();
282  while(reg != 0){
283  if (reg->width() > 3 && reg->height() > 3) {
284  vector2f p(-reg->cen_y,-reg->cen_x);
285  m.loc = p - cen;
286  m.id = marker_color_ids[c];
287  used.use(m.id.v);
288  m.area = reg->area;
289  m.angle = angle_pos(m.loc.angle());
290  m.dist = m.loc.length();
291  markers.push_back(m);
292  reg = reg->next;
293  }
294  }
295  }
296 
297  std::sort(markers.begin(),markers.end(),LessMarkerAngle());
298 
299  int num_markers = markers.size();
300  vector2f loc_sum;
301  loc_sum.zero();
302  pattern_t pattern = 0x00;
303  for(int i=0; i<num_markers; i++){
304  int j = (i+1) % num_markers;
305  markers[i].next_dist = dist(markers[i].loc,markers[j].loc);
306  markers[i].next_angle_dist = angle_pos(angle_diff(markers[i].angle,markers[j].angle));
307  loc_sum += markers[i].loc;
308  pattern = (pattern << 8) | (markers[i].id.v);
309  }
310 
311  Pattern &p = patterns[idx];
312 
313  p.setEnabled(true);
314  p.copyMarkers(markers);
315  p.marker_mean = loc_sum / max(num_markers,1);
316  p.pattern = pattern;
317  p.height = height;
318  p.robot_id = idx;
319 
320  //TODO: a nice feature would be to automatically calculate histogram
321  // percentages here.
322 
323  return(true);
324 }

387 bool MultiPatternModel::findPattern(PatternDetectionResult & result, Marker * markers,int num_markers, const PatternFitParameters & fit_params,const CameraParameters& camera_params) const {
388  if(markers==0 || num_markers<0) return(false);
389 
390  int best_idx = -1;
391  int best_ofs = 0;
392  double best_sse = sq(fit_params.fit_max_error);
393 
394  for(int ofs=0; ofs<num_markers; ofs++){
395  // calculate pattern code
396  pattern_t pattern = 0x00;
397  for(int i=0; i<num_markers; i++){
398  int j = (i + ofs) % num_markers;
399  pattern = (pattern << 8) | markers[j].id.v;
400  }
401 
402  // find covers with matching pattern code and number of markers
403  for(int i=0; i<num_patterns; i++){
404  if (patterns[i].enabled) {
405  const Pattern &p = patterns[i];
406  if(p.num_markers==num_markers && p.pattern==pattern){
407  // calculate fit error for matching pattern
408  double sse = calcFitError(p.markers,markers,num_markers,ofs,fit_params);
409 // printf("SSE: %f\n",sse);
410  /*if(unlikely(verbose > 1)){
411  float conf = SSEVsUniform(sse,fit_variance,fit_uniform);
412  printf(" 0x%04X id=%X err=%6.3f conf=%0.6f\n",
413  pattern,rc.robot_id,sqrt(sse),conf);
414  }*/
415  if(sse < best_sse){
416  best_idx = i;
417  best_ofs = ofs;
418  best_sse = sse;
419  }
420  }
421  }
422  }
423  }
424 
425  if(best_idx >= 0){
426  const Pattern &p = patterns[best_idx];
427 
428  // fix height of markers
429  for(int i=0; i<num_markers; i++){
430  vector2d marker_img_center(markers[i].reg->cen_x,markers[i].reg->cen_y);
431  vector3d marker_center3d;
432  camera_params.image2field(marker_center3d,marker_img_center,markers[i].height);
433  markers[i].loc.set(marker_center3d.x,marker_center3d.y);
434  }
435 
436  // rearrange vision markers so that the order matches the pattern model
437  Marker tmp[MaxMarkers];
438  roll(markers,tmp,num_markers,num_markers-best_ofs);
439 
440  // get the mean location of markers
441  vector2f cen_avg;
442  cen_avg.zero();
443  for(int i=0; i<num_markers; i++){
444  cen_avg += markers[i].loc;
445  }
446  cen_avg /= num_markers;
447 
448  // calculate orientation
449  vector2f orient;
450  orient.zero();
451  for(int i=0; i<num_markers; i++){
452  for(int j=0; j<i; j++){
453  vector2f vo = markers[i].loc - markers[j].loc;
454  vector2f dir = (p.markers[i].loc - p.markers[j].loc).norm();
455  vector2f o = dir.project_in(vo);
456  orient += o;
457  }
458  }
459  float angle = orient.angle();
460  orient.normalize();
461 
462  // fix bias in mean marker position
463  cen_avg -= orient.project_out(p.marker_mean);
464 
465  // save results
466  result.id = patterns[best_idx].robot_id;
467  result.idx = best_idx;
468  result.loc = cen_avg;
469  result.angle = angle;
470  result.conf = SSEVsUniform(best_sse,fit_params.fit_variance,fit_params.fit_uniform);
471  return true;
472  } else {
473  result.reset();
474  return false;
475  }
476 }
