32  if (list_entries.size() != static_cast<size_t>(6)) {

36  VarString* name = NULL;

39  VarDouble* p2_x = NULL;
40  VarDouble* p2_y = NULL;
41  VarDouble* thickness = NULL;
42  for (size_t i = 0; i < list_entries.size(); ++i) {
43  VarType* entry = list_entries[i];
44  if (entry->getName().compare("Name") == 0) {
45  name = reinterpret_cast<VarString*>(entry);
46  } else if (entry->getName().compare("P1.x") == 0) {
47  p1_x = reinterpret_cast<VarDouble*>(entry);
48  } else if (entry->getName().compare("P1.y") == 0) {
49  p1_y = reinterpret_cast<VarDouble*>(entry);
50  } else if (entry->getName().compare("P2.x") == 0) {
51  p2_x = reinterpret_cast<VarDouble*>(entry);
52  } else if (entry->getName().compare("P2.y") == 0) {
53  p2_y = reinterpret_cast<VarDouble*>(entry);
54  } else if (entry->getName().compare("Line thickness") == 0) {
55  thickness = reinterpret_cast<VarDouble*>(entry);
56  } else {
57  // Unexpected entry encountered: fail.
58  return NULL;
59  }
60  }
61  return (new FieldLine(name, p1_x, p1_y, p2_x, p2_y, thickness, list));

406  for (size_t i = 0; i < field_arc_entries.size(); ++i) {
