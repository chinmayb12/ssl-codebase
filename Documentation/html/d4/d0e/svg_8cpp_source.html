255 static int lu_factorize(float_mat &A, std::vector<int> &idx, double tol=TINY_FLOAT)
256 {
257  if ( tol <= 0.0)
258  tol = TINY_FLOAT;
259 
260  if ((A.nr_rows() == 0) || (A.nr_rows() != A.nr_cols())) {
261  //sgs_error("lu_factorize(): cannot handle empty "
262  // "or nonsquare matrices.\n");
263 
264  return 0;
265  }
266 
267  std::deque<double> scale(A.nr_rows()); // implicit pivot scaling
268  int i,j;
269  for (i = 0; i < A.nr_rows(); ++i) {
270  double maxval = 0.0;
271  for (j = 0; j < A.nr_cols(); ++j) {
272  if (fabs(A[i][j]) > maxval)
273  maxval = fabs(A[i][j]);
274  }
275  if (maxval == 0.0) {
276  //sgs_error("lu_factorize(): zero pivot found.\n");
277  return 0;
278  }
279  scale[i] = 1.0 / maxval;
280  }
281 
282  int swapNum = 1;
283  int c,r;
284  for (c = 0; c < A.nr_cols() ; ++c) { // loop over columns
285  swapNum *= partial_pivot(A, c, c, scale, idx, tol); // bring pivot to diagonal
286  for(r = 0; r < A.nr_rows(); ++r) { // loop over rows
287  int lim = (r < c) ? r : c;
288  for (j = 0; j < lim; ++j) {
289  A[idx[r]][c] -= A[idx[r]][j] * A[idx[j]][c];
290  }
291  if (r > c)
292  A[idx[r]][c] /= A[idx[c]][c];
293  }
294  }
295  permute(A,idx);
296  return swapNum;
297 }

302 static float_mat lin_solve(const float_mat &A, const float_mat &a,
303  double tol=TINY_FLOAT)
304 {
305  float_mat B(A);
306  float_mat b(a);
307  std::vector<int> idx(B.nr_rows());
308  int j;
309 
310  for (j = 0; j < B.nr_rows(); ++j) {
311  idx[j] = j; // init row swap label array
312  }
313  lu_factorize(B,idx,tol); // get the lu-decomp.
314  permute(b,idx); // sort the inhomogeneity to match the lu-decomp

316  lu_backsubst(B,b); // solve the backward problem

413 {
414  std::deque<double> res(v.size(), 0.0);
415  if ((width < 1) || (deg < 0) || (v.size() < (2 * width + 2))) {
416  //sgs_error("sgsmooth: parameter error.\n");
417  return v;
418  }
419 
420  const int window = 2 * width + 1;
421  const int endidx = v.size() - 1;
422 
423  // do a regular sliding window average
424  int i,j;
425  if (deg == 0) {
426  // handle border cases first because we need different coefficients
427 #if defined(_OPENMP)
428 #pragma omp parallel for private(i,j) schedule(static)
429 #endif
430  for (i = 0; i < width; ++i) {
431  const double scale = 1.0/double(i+1);
432  const std::deque<double> c1(width, scale);
433  for (j = 0; j <= i; ++j) {
434  res[i] += c1[j] * v[j];
435  res[endidx - i] += c1[j] * v[endidx - j];
436  }
437  }
438 
439  // now loop over rest of data. reusing the "symmetric" coefficients.
440  const double scale = 1.0/double(window);
441  const std::deque<double> c2(window, scale);
442 #if defined(_OPENMP)
443 #pragma omp parallel for private(i,j) schedule(static)
444 #endif
445  for (i = 0; i <= (v.size() - window); ++i) {
446  for (j = 0; j < window; ++j) {
447  res[i + width] += c2[j] * v[i + j];
448  }
449  }
450  return res;
451  }
452 
453  // handle border cases first because we need different coefficients
454 #if defined(_OPENMP)
455 #pragma omp parallel for private(i,j) schedule(static)
456 #endif
457  for (i = 0; i < width; ++i) {
458  std::deque<double> b1(window, 0.0);
459  b1[i] = 1.0;
460 
461  const std::deque<double> c1(sg_coeff(b1, deg));
462  for (j = 0; j < window; ++j) {
463  res[i] += c1[j] * v[j];
464  res[endidx - i] += c1[j] * v[endidx - j];
465  }
466  }

521 std::deque<double> sg_derivative(const std::deque<double> &v, const int width,
522  const int deg, const double h)
523 {
524  std::deque<double> res(v.size(), 0.0);
525  if ((width < 1) || (deg < 1) || (v.size() < (2 * width + 2))) {
526  printf("sgsderiv: parameter error.\n");
527  return v;
528  }
529 
530  const int window = 2 * width + 1;
531 
532  // handle border cases first because we do not repeat the fit
533  // lower part
534  std::deque<double> b(window, 0.0);
535  int i,j;
536 
537  for (i = 0; i < window; ++i) {
538  b[i] = v[i] / h;
539  }
540  const std::deque<double> c(lsqr_fprime(b, deg));
541  for (j = 0; j <= width; ++j) {
542  res[j] = c[j];
543  }
544  // upper part. direction of fit is reversed
545  for (i = 0; i < window; ++i) {
546  b[i] = v[v.size() - 1 - i] / h;
547  }
548  const std::deque<double> d(lsqr_fprime(b, deg));
549  for (i = 0; i <= width; ++i) {
550  res[v.size() - 1 - i] = -d[i];
551  }
552 
553  // now loop over rest of data. wasting a lot of least squares calcs
554  // since we only use the middle value.
555  #if defined(_OPENMP)
556  #pragma omp parallel for private(i,j) schedule(static)
557  #endif
558  for (i = 1; i < (v.size() - window); ++i) {
559  for (j = 0; j < window; ++j) {
560  b[j] = v[i + j] / h;
561  }
562  res[i + width] = lsqr_fprime(b, deg)[width];
563  }
564  return res;
565 }
