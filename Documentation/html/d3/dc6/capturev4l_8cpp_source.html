34 // A note about the capture format.... (from Eric)
35 // - There appears to be a different method of byte packing in the V4L
36 // library versus standard YUYV and UYUV packing used in ssl-vision
37 // http://linuxtv.org/downloads/v4l-dvb-apis/V4L2-PIX-FMT-YUYV.html
38 // - Additional some, very basic cameras only support YUYV and not UYVY,
39 // which is the expected 4:2:2 format within ssl-vision for the YUV lookup.
40 // - Thus, we make the extra hop into the common RGB space immediately
41 // at capture instead of deferring this to a convsersion later.
42 // Committs earlier in the life of this code indicate other attempts,
43 // which may be helpful to future developers.
44 // - One ramification is a slower copy/decode process. On a desktop or
45 // laptop, this doesn't seem to influce frame rate.
46 // - This is not ideal, but the problem is left to future developers
47 // to ferret out proper settings for the video format (see v4l2_format)
48 // and colorspaces within ssl-vision and V4L. Until then, it works.

127 bool GlobalV4Linstance::obtainInstance(char *szDevice_)
128 {
129  lock();
130  bool bSuccess = (pollset.fd > -1);
131  //fprintf(stderr,"WARNING: obtainInstance ENTER : counter : %d!\n", counter);
132 
133  if (counter == 0) {
134  if (pollset.fd < 0) {
135  counter = 1;
136  removeInstance(false);
137  }
138 
139  pollset.fd = open(szDevice_, O_RDWR|O_NONBLOCK);
140  if(pollset.fd > -1) bSuccess = true;
141 
142  if (bSuccess) {
143  pollset.events = POLLIN;
144 
145  // test capture capabilities
146  v4l2_capability cap;
147  if(xioctl(pollset.fd, VIDIOC_QUERYCAP, &cap, "VideoQueryCap") == 0){
148  if(!(cap.capabilities & V4L2_CAP_VIDEO_CAPTURE) ||
149  !(cap.capabilities & V4L2_CAP_STREAMING)){
150  bSuccess = false;
151  }
152  }else{
153  bSuccess = false;
154  }
155  }
156 
157  // set cropping to default (no effect if not supported)
158  if(bSuccess && false){
159  v4l2_cropcap cropcap;
160  v4l2_crop crop;
161  mzero(cropcap);
162  mzero(crop);
163  cropcap.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
164  xioctl(pollset.fd, VIDIOC_CROPCAP, &cropcap, "CropCap");
165  crop.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
166  crop.c = cropcap.defrect;
167  xioctl(pollset.fd, VIDIOC_S_CROP, &crop, "SetCrop");
168  }
169  if (!bSuccess ) {
170  if(pollset.fd > -1) close(pollset.fd);
171  pollset.fd = -1;
172  }
173  else {
174  strncpy(szDevice, szDevice_, 128);
175  }
176  }
177  if (bSuccess)
178  counter++;
179  //fprintf(stderr,"WARNING: obtainInstance EXIT : counter : %d!\n", counter);
180 
181  unlock();
182  return bSuccess;
183 }

214 int GlobalV4Linstance::xioctl(int fd,int request,void *data,
215  const char *error_str)
216 {
217  // try the ioctl, which should succeed in the common case
218  int ret = ioctl(fd,request,data);
219  if(ret >= 0) return(ret);
220 
221  // retry if we were interrupted (up to a few times)
222  int n=0;
223  while(ret!=0 && errno==EINTR && n<8){
224  ret = ioctl(fd,request,data);
225  n++;
226  }
227 
228  // report error
229  if(ret != 0 && error_str) {
230  fprintf(stderr,"GlobalV4Linstance: %s returned %d (%s)\n",
231  error_str,ret,strerror(errno));
232  }
233 
234  return(ret);

281 const GlobalV4Linstance::image_t *GlobalV4Linstance::captureFrame(int iMaxSpin)
282 {
283  if(!waitForFrame(300)) {
284  fprintf(stderr,"GlobalV4Linstance: error waiting for frame '%s'\n", szDevice);
285  return(NULL);
286  }
287 
288  do {
289  // get the frame
290  dequeueBuffer(tempbuf);
291 
292  // poll to see if a another frame is already available
293  // if so, break out now
294  if(!waitForFrame(0)) break;
295 
296  // otherwise, drop this frame
297  enqueueBuffer(tempbuf);
298  }
299  while(iMaxSpin--);
300 
301  int i = tempbuf.index;
302  img[i].timestamp = tempbuf.timestamp;
303  img[i].field = (tempbuf.field == V4L2_FIELD_BOTTOM);
304  return(&(img[i]));

333 bool GlobalV4Linstance::startStreaming(int iWidth_, int iHeight_, int iInputIdx)
334 {
335  struct v4l2_requestbuffers req;
336 
337  // Set video format
338  v4l2_format fmt;
339  mzero(fmt);
340  fmt.fmt.pix.width = iWidth_;
341  fmt.fmt.pix.height = iHeight_;
342  fmt.fmt.pix.pixelformat = V4L2_PIX_FMT_YUYV; //see note at top of file
343  fmt.fmt.pix.field = V4L2_FIELD_ALTERNATE;
344  fmt.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
345  if (!xioctl(VIDIOC_S_FMT, &fmt, "SetFormat")) {
346  printf("Warning: Could not set format, '%s'; was device previously started?\n", szDevice);
347  // return(false);
348  }
349 
350  // Set Input and Controls (for more advanced v4l as well)
351  // http://www.linuxtv.org/downloads/v4l-dvb-apis/vidioc-g-input.html
352  // vid.setInput(0); // main capture (cable or camera video)
353  // vid.setInput(1); // Component video
354  // vid.setInput(2); // S-video
355  xioctl(VIDIOC_S_INPUT,&iInputIdx,"SetInput");
356  /*
357  v4l2_std_id id = V4L2_STD_NTSC_M; //TODO: modify/detect for other cameras
358  xioctl(VIDIOC_S_STD,&id,"SetStandard");
359 
360  long lControlVal = static_cast<long>(0.5f*(1 << 16));
361  setControl(V4L2_CID_BRIGHTNESS, lControlVal);
362  setControl(V4L2_CID_HUE, lControlVal);
363  setControl(V4L2_CID_CONTRAST, lControlVal);
364  lControlVal = static_cast<long>(0.9f*(1 << 16));
365  setControl(V4L2_CID_SATURATION, lControlVal);
366  */
367 
368  // Request mmap-able capture buffers
369  mzero(req);
370  req.count = V4L_STREAMBUFS;
371  req.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
372  req.memory = V4L2_MEMORY_MMAP;
373  if(!xioctl(VIDIOC_REQBUFS, &req, "Request Buffers") || req.count != V4L_STREAMBUFS) {
374  printf("REQBUFS returned error, count %d\n", req.count);
375  return(false);
376  }
377 
378  // set up individual buffers
379  mzero(img,V4L_STREAMBUFS);
380  mzero(tempbuf);
381  tempbuf.type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
382  tempbuf.memory = V4L2_MEMORY_MMAP;
383 
384  for(unsigned i=0; i<req.count; i++){
385  tempbuf.index = i;
386  if(!xioctl(VIDIOC_QUERYBUF, &tempbuf, "Allocate query buffer")) {
387  printf("QUERYBUF returned error, '%s'\n", szDevice);
388  return(false);
389  }
390  img[i].index = i;
391  img[i].length = tempbuf.length;
392  img[i].data = static_cast<unsigned char*>(
393  mmap(NULL, tempbuf.length, PROT_READ | PROT_WRITE, MAP_SHARED,
394  pollset.fd, tempbuf.m.offset));
395 
396  if(img[i].data == MAP_FAILED){
397  printf("mmap() returned error %d (%s)\n",errno,strerror(errno));
398  return(false);
399  }
400  }
401 
402  //enqueue buffer that we just memmapped/allocated
403  for(unsigned i=0; i<req.count; i++){
404  tempbuf.index = i;
405  if(!enqueueBuffer(tempbuf)){
406  printf("Error queueing initial buffers, '%s'\n", szDevice);
407  return(false);
408  }
409  }
410 
411  //start actual stream
412  v4l2_buf_type type = V4L2_BUF_TYPE_VIDEO_CAPTURE;
413  return xioctl(VIDIOC_STREAMON, &type, "StreamOn");
414 }
