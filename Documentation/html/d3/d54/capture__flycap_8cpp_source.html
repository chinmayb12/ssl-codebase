51 ColorFormat pixelFormatToColorFormat(PixelFormat pixel_format)
52 {
53  ColorFormat format;
54  //fprintf(stderr, "Pixel Format: %i\n",(int)pixel_format);
55  switch(pixel_format) {
56  case PIXEL_FORMAT_MONO8: format = COLOR_MONO8; break;
57  case PIXEL_FORMAT_411YUV8: format = COLOR_YUV411; break;//< YUV 4:1:1.
58  case PIXEL_FORMAT_444YUV8: format = COLOR_YUV444; break;//< YUV 4:4:4.
59  case PIXEL_FORMAT_RGB8: format = COLOR_RGB8; break;//< R = G = B = 8 bits.
60  case PIXEL_FORMAT_MONO16: format = COLOR_MONO16; break;//< 16 bits of mono information.
61  case PIXEL_FORMAT_RGB16: format = COLOR_RGB16; break;//< R = G = B = 16 bits.
62  case PIXEL_FORMAT_RAW8: format = COLOR_RAW8; break;//< 8 bit raw data output of sensor.
63  case PIXEL_FORMAT_RAW16: format = COLOR_RAW16; break;//< 16 bit raw data output of sensor.
64  case PIXEL_FORMAT_422YUV8: format = COLOR_YUV422_UYVY; break;
65  case UNSPECIFIED_PIXEL_FORMAT: format = COLOR_UNDEFINED; break; //< Unspecified pixel format.
66  default: format = COLOR_UNDEFINED; break;
67  }
68  //fprintf(stderr,"Color format: %i\n",(int)format);
69 
70  return format;
71 }

523 bool CaptureFlycap::convertFrame(const RawImage & src,
524  RawImage & target,
525  ColorFormat output_fmt,
526  bool debayer,
527  dc1394color_filter_t bayer_format,
528  dc1394bayer_method_t bayer_method,
529  int y16bits) {
530  mutex.lock();
531 
532  int width = v_width->getInt();
533  int height = v_height->getInt();
534 
535  ColorFormat src_fmt=src.getColorFormat();
536  if (target.getData()==0) {
537  //allocate target, if it does not exist yet
538  target.allocate(output_fmt,src.getWidth(),src.getHeight());
539  } else {
540  target.ensure_allocation(output_fmt,src.getWidth(),src.getHeight());
541  }
542  target.setTime(src.getTime());
543  if (output_fmt==src_fmt) {
544  //just do a memcpy
545  memcpy(target.getData(),src.getData(),src.getNumBytes());
546  } else {
547  //do some more fancy conversion
548  if ((src_fmt==COLOR_MONO8 || src_fmt==COLOR_RAW8) && output_fmt==COLOR_RGB8) {
549  //check whether to debayer or simply average to a grey rgb image
550  if (debayer) {
551  //de-bayer
552  if ( dc1394_bayer_decoding_8bit( src.getData(), target.getData(), src.getWidth(), src.getHeight(), bayer_format, bayer_method) != DC1394_SUCCESS ) {
553  #ifndef VDATA_NO_QT
554  mutex.unlock();
555  #endif
556  return false;
557  }
558  } else {
559  dc1394_convert_to_RGB8(src.getData(),target.getData(), width, height, 0,
560  DC1394_COLOR_CODING_MONO8, 8);
561  //Conversions::y2rgb (src.getData(), target.getData(), src.getNumPixels());
562  }
563  } else if ((src_fmt==COLOR_MONO16 || src_fmt==COLOR_RAW16)) {
564  //check whether to debayer or simply average to a grey rgb image
565  if (debayer && output_fmt==COLOR_RGB16) {
566  //de-bayer
567  if ( dc1394_bayer_decoding_16bit( (uint16_t *)src.getData(), (uint16_t *)target.getData(), src.getWidth(), src.getHeight(), bayer_format, bayer_method, y16bits) != DC1394_SUCCESS ) {
568  fprintf(stderr,"Error in 16bit Bayer Conversion");
569 
570  #ifndef VDATA_NO_QT
571  mutex.unlock();
572  #endif
573  return false;
574  }
575  } else if (debayer==false && output_fmt==COLOR_RGB8) {
576 
577  dc1394_convert_to_RGB8(src.getData(),target.getData(), width, height, 0,
578  ((src_fmt==COLOR_MONO16) ? DC1394_COLOR_CODING_MONO16 : DC1394_COLOR_CODING_RAW16), y16bits);
579  //Conversions::y162rgb (src.getData(), target.getData(), src.getNumPixels(), y16bits);
580  } else {
581  fprintf(stderr,"Cannot copy and convert frame...unknown conversion selected from: %s to %s\n",Colors::colorFormatToString(src_fmt).c_str(),Colors::colorFormatToString(output_fmt).c_str());
582  mutex.unlock();
583  return false;
584  }
585  } else if (src_fmt==COLOR_YUV411 && output_fmt==COLOR_RGB8) {
586  dc1394_convert_to_RGB8(src.getData(),target.getData(), width, height, 0,
587  DC1394_COLOR_CODING_YUV411, 8);
588  //Conversions::uyyvyy2rgb (src.getData(), target.getData(), src.getNumPixels());
