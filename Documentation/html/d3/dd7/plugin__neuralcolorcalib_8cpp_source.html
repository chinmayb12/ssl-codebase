32 int PluginNeuralColorCalib::TrainNeuralopenCV() {
33  CvANN_MLP_TrainParams netparams;
34  CvTermCriteria term_crit;
35  double * training;
36  training= new double [n_samples * sizeInputLayer * sizeInputLayer];
37  double * target;
38  target= new double [n_samples * sizeOutLayer * sizeOutLayer];
39  CvMat trainin;
40  CvMat trainout;
41  int interactions_ran=0;
42 
43  term_crit.max_iter =1000; //maximum number of epochs to train
44  term_crit.epsilon = 1e-5; //maximum tolerance to errors of the network during training
45  term_crit.type = CV_TERMCRIT_ITER + CV_TERMCRIT_EPS; //indicates training stops either when performance or max. epochs is reached
46  netparams=CvANN_MLP_TrainParams(term_crit, CvANN_MLP_TrainParams::BACKPROP,0.1,0.1); //specifies learning algorithm, learning rate and momentum
47 
48  for (unsigned int n=0; n<neuro_trainingset->size(); n++){
49  training[n] = neuro_trainingset->at(n);
50  }
51  for (unsigned int n=0; n<neuro_targset->size(); n++){
52  target[n] = neuro_targset->at(n);
53  }
54 
55  trainin = cvMat(n_samples,3,CV_64F,training);
56  trainout = cvMat(n_samples,sizeOutLayer,CV_64F,target);
57  interactions_ran = neuronet->train (&trainin, &trainout,0,0,netparams,0); //call training functions of the object
58  isNetSet=true;
59  return interactions_ran;
60

62 }

64 int PluginNeuralColorCalib::RunNeuralopenCV(double *input) {
65  double tmpout=0,maxout=0;
66  int noutput=0;
67 
68  if (isNetSet){ //if network is not trained do nothing
69  cvSetData(realinput,input,sizeof(double)*3); //inserts the input into the OpenCV matrix format
70  neuronet->predict (realinput,netout); //call prediction (ask the output) of the object for a given input.
71 
72  for (int j=0; j<sizeOutLayer; j++){ //inserts the output in an output array
73  tmpout=CV_MAT_ELEM(*netout,double,0,j);
74  if (tmpout > maxout){
75  noutput=j;
76  maxout=tmpout;
77  }
78  }
79 
80  return noutput;
81  }
82  else
83  return -1;
84 }

159 void PluginNeuralColorCalib::mouseEvent( QMouseEvent * event, pixelloc loc) {
160  QTabWidget* tabw = (QTabWidget*) lutw->parentWidget()->parentWidget();
161  if (tabw->currentWidget() == lutw) {
162  if (event->buttons()==Qt::LeftButton && lutw->getStateChannel() != -1) {
163  FrameBuffer * rb=getFrameBuffer();
164  if (rb!=0) {
165  rb->lockRead();
166  int idx=rb->curRead();
167  FrameData * frame = rb->getPointer(idx);
168  if (loc.x < frame->video.getWidth() && loc.y < frame->video.getHeight() && loc.x >=0 && loc.y >=0) {
169  if (frame->video.getWidth() > 1 && frame->video.getHeight() > 1) {
170  yuv color;
171  //if converting entire image then blanking is not needed
172  ColorFormat source_format=frame->video.getColorFormat();
173  if (source_format==COLOR_RGB8) {
174  //plain copy of data
175  rgbImage img(frame->video);
176  color=Conversions::rgb2yuv(img.getPixel(loc.x,loc.y));
177  } else if (source_format==COLOR_YUV444) {
178  yuvImage img(frame->video);
179  color=img.getPixel(loc.x,loc.y);
180  } else if (source_format==COLOR_YUV422_UYVY) {
181  uyvy color2 = *((uyvy*)(frame->video.getData() + (sizeof(uyvy) * (((loc.y * (frame->video.getWidth())) + loc.x) / 2))));
182  color.u=color2.u;
183  color.v=color2.v;
184  if ((loc.x % 2)==0) {
185  color.y=color2.y1;
186  } else {
187  color.y=color2.y2;
188  }
189  } else {
190  //blank it:
191  fprintf(stderr,"Unable to pick color from frame of format: %s\n",Colors::colorFormatToString(source_format).c_str());
192  fprintf(stderr,"Currently supported are rgb8, yuv444, and yuv422 (UYVY).\n");
193  fprintf(stderr,"(Feel free to add more conversions to plugin_neuralcolorcalib.cpp).\n");
194  }
195 
196  neuro_trainingset->push_back( (double)color.y );
197  neuro_trainingset->push_back( (double)color.u );
198  neuro_trainingset->push_back( (double)color.v );
199 
200  for (int j=0; j < sizeOutLayer; j++){
201  if (lutw->getStateChannel() == j)
202  neuro_targset->push_back( lutw->getStateChannel() );
203  else
204  neuro_targset->push_back( 0 );
205  }
206  this->n_samples++;
207  lutw->setSamplesLabel(this->n_samples);
208 
209  }
210  }
211  rb->unlockRead();
212  }
213  event->accept();
214  }
215 
216  }
217  else
218  event->ignore();
219 }

221 void PluginNeuralColorCalib::CopyNeuraltoLUT() {
222 
223  double input[3];
224  int output;
225 
226  if (isNetSet)
227  {
228  lut->lock();
229  for (int y = 0; y <= 255; y+= (0x1 << lut->X_SHIFT)) {
230  for (int u = 0; u <= 255; u+= (0x1 << lut->Y_SHIFT)) {
231  for (int v = 0; v <= 255; v+= (0x1 << lut->Z_SHIFT)) {
232  input[0]= (double)y;
233  input[1]= (double)u;
234  input[2]= (double)v;
235  output = RunNeuralopenCV(input);
236  if (output== -1)
237  output = 0;
238  lut->set(y,u,v,output); //Writes values to the camera's LUT table
239  }
240  }
241  }
242  lut->unlock();
243  }
