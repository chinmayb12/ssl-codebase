82 bool CaptureFromFile::startCapture()
83 {
84 #ifndef VDATA_NO_QT
85  mutex.lock();
86 #endif
87  if(images.size() == 0)
88  {
89  // Acquire a list of file names
90  DIR *dp;
91  struct dirent *dirp;
92  if((v_cap_dir->getString() == "") || ((dp = opendir(v_cap_dir->getString().c_str())) == 0))
93  {
94  fprintf(stderr,"Failed to open directory %s \n", v_cap_dir->getString().c_str());
95 #ifndef VDATA_NO_QT
96  mutex.unlock();
97 #endif
98  is_capturing=false;
99  return false;
100  }
101  while ((dirp = readdir(dp)))
102  {
103  if (strcmp(dirp->d_name,".") != 0 && strcmp(dirp->d_name,"..") != 0)
104  {
105  if(isImageFileName(std::string(dirp->d_name)))
106  imgs_to_load.push_back(v_cap_dir->getString() + std::string(dirp->d_name));
107  else
108  fprintf(stderr,"Not a valid image file: %s \n", dirp->d_name);
109  }
110  }
111  closedir(dp);
112  if(imgs_to_load.size() == 0)
113  {
114 #ifndef VDATA_NO_QT
115  mutex.unlock();
116 #endif
117  is_capturing=false;
118  return false;
119  }
120 
121  // Read images to buffer in memory:
122  imgs_to_load.sort();
123  imgs_it = imgs_to_load.begin();
124  std::list<std::string>::iterator currentImage = imgs_it;
125  while(currentImage != imgs_to_load.end())
126  {
127  int width(-1);
128  int height(-1);
129  rgba* rgba_img = ImageIO::readRGBA(width, height, currentImage->c_str());
130  fprintf (stderr, "Loaded %s \n", currentImage->c_str());
131  images.push_back(rgba_img);
132  heights.push_back(height);
133  widths.push_back(width);
134  ++currentImage;
135  }
136  currentImageIndex = 0;
137  }
138  is_capturing=true;
139 
140 #ifndef VDATA_NO_QT
141  mutex.unlock();
142 #endif
143  return true;
144 }

146 bool CaptureFromFile::isImageFileName(const std::string& fileName)
147 {
148  // Get ending and turn it to uppercase:
149  string::size_type pointPos = fileName.find_last_of(".");
150  if(pointPos == string::npos)
151  return false;
152  string ending = fileName.substr(pointPos+1);
153  for(unsigned int i=0; i<ending.size(); ++i)
154  ending[i] = toupper(ending[i]);
155  // Compare against list of valid endings
156  for(unsigned int i=0; i<validImageFileEndings.size();++i)
157  if(ending == validImageFileEndings[i])
158  return true;
159  return false;
160 }

162 bool CaptureFromFile::copyAndConvertFrame(const RawImage & src, RawImage & target)
163 {
164 #ifndef VDATA_NO_QT
165  mutex.lock();
166 #endif
167  ColorFormat output_fmt = Colors::stringToColorFormat(v_colorout->getSelection().c_str());
168  ColorFormat src_fmt=src.getColorFormat();
169 
170  if (target.getData()==0)
171  target.allocate(output_fmt, src.getWidth(), src.getHeight());
172  else
173  target.ensure_allocation(output_fmt, src.getWidth(), src.getHeight());
174 
175  target.setTime(src.getTime());
176 
177  if (output_fmt == src_fmt)
178  {
179  if (src.getData() != 0)
180  memcpy(target.getData(),src.getData(),src.getNumBytes());
181  }
182  else if (src_fmt == COLOR_RGB8 && output_fmt == COLOR_YUV422_UYVY)
183  {
184  if (src.getData() != 0)
185  dc1394_convert_to_YUV422(src.getData(), target.getData(), src.getWidth(), src.getHeight(),
186  DC1394_BYTE_ORDER_UYVY, DC1394_COLOR_CODING_RGB8, 8);
187  }
188  else if (src_fmt == COLOR_YUV422_UYVY && output_fmt == COLOR_RGB8)
189  {
190  if (src.getData() != 0)
191  dc1394_convert_to_RGB8(src.getData(),target.getData(), src.getWidth(), src.getHeight(),
192  DC1394_BYTE_ORDER_UYVY, DC1394_COLOR_CODING_YUV422, 8);
193  }
194  else
195  {
196  fprintf(stderr,"Cannot copy and convert frame...unknown conversion selected from: %s to %s\n",
197  Colors::colorFormatToString(src_fmt).c_str(),
198  Colors::colorFormatToString(output_fmt).c_str());
199 #ifndef VDATA_NO_QT
200  mutex.unlock();
201 #endif
202  return false;
203  }
204 #ifndef VDATA_NO_QT
205  mutex.unlock();
206 #endif
207  return true;
208 }

210 RawImage CaptureFromFile::getFrame()
211 {
212 #ifndef VDATA_NO_QT
213  mutex.lock();
214 #endif
215 
216  RawImage result;
217  result.setColorFormat(COLOR_RGB8);
218  result.setTime(0.0);
219  rgba* rgba_img = 0;
220  int width;
221  int height;
222  if(images.size())
223  {
224  rgba_img = images[currentImageIndex];
225  width = widths[currentImageIndex];
226  height = heights[currentImageIndex];
227  currentImageIndex = (currentImageIndex + 1) % images.size();
228  }
229  if (rgba_img == 0)
230  {
231  fprintf (stderr, "CaptureFromFile Error, no images available");
232  is_capturing=false;
233  result.setData(0);
234  result.setWidth(640);
235  result.setHeight(480);
236  frame = 0;
237  }
238  else
239  {
240  frame = new unsigned char[width*height*3];
241  unsigned char* p = &frame[0];
242  for (int i=0; i < width * height; i++)
243  {
244  *p = rgba_img[i].r;
245  p++;
246  *p = rgba_img[i].g;
247  p++;
248  *p = rgba_img[i].b;
249  p++;
250  }
251  result.setWidth(width);
252  result.setHeight(height);
253  result.setData(frame);
254  timeval tv;
255  gettimeofday(&tv,0);
256  result.setTime((double)tv.tv_sec + tv.tv_usec*(1.0E-6));
257  }
258 #ifndef VDATA_NO_QT
259  mutex.unlock();
260 #endif
261  return result;
262 }
