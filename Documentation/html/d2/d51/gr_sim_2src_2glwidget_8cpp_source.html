284 void GLWidget::update3DCursor(int mouse_x,int mouse_y)
285 {
286  if (!ssl->g->isGraphicsEnabled()) return;
287  ssl->updatedCursor = true;
288  dVector3 xyz,hpr;
289  dReal fx,fy,fz,rx,ry,rz,ux,uy,uz,px,py,pz;
290  ssl->g->getViewpoint(xyz,hpr);
291  ssl->g->getCameraForward(fx,fy,fz);
292  ssl->g->getCameraRight(rx,ry,rz);
293  ux = ry*fz - rz*fy;
294  uy = rz*fx - rx*fz;
295  uz = rx*fy - ry*fx;
296  dReal w = width();
297  dReal h = height();
298  dReal xx,yy,z;
299  dReal x = 1.0f - 2.0f*(dReal) mouse_x / w;
300  dReal y = 1.0f - 2.0f*(dReal) mouse_y / h;
301  ssl->g->getFrustum(xx,yy,z);
302  x *= xx;
303  y *= yy;
304  px = -ux*y - rx*x - z*fx;
305  py = -uy*y - ry*x - z*fy;
306  pz = -uz*y - rz*x - z*fz;
307  ssl->ray->setPose(xyz[0],xyz[1],xyz[2],px,py,pz);
308 }

345 void GLWidget::step()
346 {
347  static double lastBallSpeed=-1;
348  const dReal* ballV = dBodyGetLinearVel(ssl->ball->body);
349  double ballSpeed = ballV[0]*ballV[0] + ballV[1]*ballV[1] + ballV[2]*ballV[2];
350  ballSpeed = sqrt(ballSpeed);
351  lastBallSpeed = ballSpeed;
352  rendertimer.restart();
353  m_fps = frames /(time.elapsed()/1000.0);
354  if (!(frames % ((int)(ceil(cfg->DesiredFPS()))))) {
355  time.restart();
356  frames = 0;
357  }
358  if (first_time) {ssl->step();first_time = false;}
359  else {
360  if (cfg->SyncWithGL())
361  {
362  dReal ddt=rendertimer.elapsed()/1000.0;
363  if (ddt>0.05) ddt=0.05;
364  ssl->step(ddt);
365  }
366  else {
367  ssl->step(cfg->DeltaTime());
368  }
369  }
370  frames ++;
371 }
