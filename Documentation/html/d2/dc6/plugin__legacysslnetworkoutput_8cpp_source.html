35 bool PluginLegacySSLNetworkOutput::DoubleSizeToSingleSize(
36  const bool primary_field,
37  const float x_in,
38  const float y_in,
39  const float a_in,
40  float* x_out,
41  float* y_out,
42  float* a_out) {
43  const float half_field_length =
44  primary_field ?
45  -min(GetFieldLine("TopTouchLine", _field.field_lines)->
46  p1_x->getDouble(),
47  GetFieldLine("TopTouchLine", _field.field_lines)->
48  p2_x->getDouble()) :
49  max(GetFieldLine("TopTouchLine", _field.field_lines)->p1_x->getDouble(),
50  GetFieldLine("TopTouchLine", _field.field_lines)->p2_x->getDouble());
51  _field.field_length->getDouble();
52  const float line_width =
53  GetFieldLine("HalfwayLine", _field.field_lines)->thickness->getDouble();
54  // Ignore robots not within the referee margin minus robot radius
55  if (primary_field) {
56  if (x_in > 425.0 - 90.0) return false;
57  *y_out = -(x_in + 0.5 * half_field_length - 0.5 * line_width);
58  } else {
59  if (x_in < -425.0 + 90.0) return false;
60  *y_out = -(x_in - 0.5 * half_field_length + 0.5 * line_width);
61  }
62  *x_out = y_in;
63  *a_out = angle_mod(a_in - RAD(90.0));
64  return true;
65 }
