50  int sobel = (-1*img.getPixel(x-1,y-1).v) +

135  //Scans img vertical line from pixel (x,yStart) to (x,yEnd) and returns the y coordinate of the center of the horizontal line
136  //The line is demarcated by the low-to-high transition maxBrightEdge, and the high-to-low transition maxDarkEdge

191  //Scans img along the line from (xStart,yStart) to (xEnd,yEnd) and returns the point p of the center of the line
192  static void centerOfLine(const greyImage& img, int xStart, int xEnd, int yStart, int yEnd, GVector::vector2d<double> &p, bool &centerFound,
193  int threshold)
194  {
195  int maxDarkEdge(0);
196  int maxBrightEdge(0);
197  GVector::vector2d<double> maxDarkP(0.0,0.0);
198  GVector::vector2d<double> maxBrightP(0.0,0.0);
199  int numSteps = max(abs(xEnd-xStart),abs(yEnd-yStart));
200  double xIncr = ((double) (xEnd-xStart))/((double) numSteps);
201  double yIncr = ((double) (yEnd-yStart))/((double) numSteps);
202 
203  for(int i=0; i<=numSteps; i++)
204  {
205  int x = floor((double) xStart + ((double) i)*xIncr + 0.5);
206  int y = floor((double) yStart + ((double) i)*yIncr + 0.5);
207  if (!validImagePixel(img, x, y)) continue;
208  int Dx = Sobel::horizontalDarker(img, x, y, threshold);
209  int Dy = Sobel::verticalDarker(img, x, y, threshold);
210  int Bx = Sobel::horizontalBrighter(img, x, y, threshold);
211  int By = Sobel::verticalBrighter(img, x, y, threshold);
212  int currentDarkEdge = (Dx*Dx+Dy*Dy);
213  int currentBrightEdge = (Bx*Bx+By*By);
214  if( currentDarkEdge > maxDarkEdge)
215  {
216  maxDarkEdge = currentDarkEdge;
217  maxDarkP = GVector::vector2d<double>(x,y);
218  }
219  if(currentBrightEdge > maxBrightEdge)
220  {
221  maxBrightEdge = currentBrightEdge;
222  maxBrightP = GVector::vector2d<double>(x,y);
223  }
224  }
225  p = 0.5*(maxBrightP + maxDarkP);
226  centerFound = (maxDarkEdge>0) && (maxBrightEdge>0);
227  }
