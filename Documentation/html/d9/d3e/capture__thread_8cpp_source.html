122 void CaptureThread::selectCaptureMethod() {
123  capture_mutex.lock();
124  CaptureInterface * old_capture=capture;
125  CaptureInterface * new_capture=0;
126  if(captureModule->getString() == "Read from files") {
127  new_capture = captureFiles;
128  } else if(captureModule->getString() == "Generator") {
129  new_capture = captureGenerator;
130 #ifdef MVIMPACT
131  } else if(captureModule->getString() == "BlueFox2") {
132  new_capture = captureBlueFox2;
133 #endif
134  } else if(captureModule->getString() == "Video 4 Linux") {
135  new_capture = captureV4L;
136  } else if(captureModule->getString() == "DC 1394") {
137  new_capture = captureDC1394;
138 #ifdef FLYCAP
139  } else if(captureModule->getString() == "Flycapture") {
140  new_capture = captureFlycap;
141 #endif
142  }
143 
144  if (old_capture!=0 && new_capture!=old_capture && old_capture->isCapturing()) {
145  capture_mutex.unlock();
146  stop();
147  capture_mutex.lock();
148  capture=new_capture;
149  capture_mutex.unlock();
150  return;
151  }
152  capture=new_capture;
153  capture_mutex.unlock();
154 }

203 void CaptureThread::run() {
204  CaptureStats * stats;
205  bool changed;
206 
207  if (affinity!=0) {
208  affinity->demandCore(camId);
209  }
210 
211  while(true) {
212  if (rb!=0) {
213  int idx=rb->curWrite();
214  FrameData * d=rb->getPointer(idx);
215  if ((stats=(CaptureStats *)d->map.get("capture_stats")) == 0) {
216  stats=(CaptureStats *)d->map.insert("capture_stats",new CaptureStats());
217  }
218  capture_mutex.lock();
219  if ((capture != 0) && (capture->isCapturing())) {
220  RawImage pic_raw=capture->getFrame();
221  d->time=pic_raw.getTime();
222  bool bSuccess = capture->copyAndConvertFrame( pic_raw,d->video);
223  capture_mutex.unlock();
224 
225  if (bSuccess) { //only on a good frame read do we proceed
226  counter->count();
227  stats->total=d->number=counter->getTotal();
228  d->cam_id=camId;
229  stats->fps_capture=counter->getFPS(changed);
230 
231  stack_mutex.lock();
232  if (stack!=0) {
233  stack->process(d);
234  stack->postProcess(d);
235  }
236  stack_mutex.unlock();
237  rb->nextWrite(true);
238 
239 
240  if (changed) {
241  if (c_auto_refresh->getBool()==true) {
242  capture_mutex.lock();
243  if ((capture != 0) && (capture->isCapturing())) capture->readAllParameterValues();
244  capture_mutex.unlock();
245  }
246  stack_mutex.lock();
247  stack->updateTimingStatistics();
248  stack_mutex.unlock();
249  }
250  capture_mutex.lock();
251  if ((capture != 0) && (capture->isCapturing())) {
252  capture->releaseFrame();
253  }
254  capture_mutex.unlock();
255  }
256  } else {
257  stats->total=d->number=counter->getTotal();
258  stats->fps_capture=counter->getFPS(changed);
259  //we are not capturing...chill this thread out...
260  capture_mutex.unlock();
261  usleep(5000);
262  }
263  if (_kill) {
264  capture_mutex.lock();
265  if(capture != 0) {
266  capture->stopCapture();
267  //make sure to read latest params from camera to be saved to file...
268  if (capture->isCapturing()) capture->readAllParameterValues();
269  }
270  capture_mutex.unlock();
271  return;
272  }
273  }
274  }
