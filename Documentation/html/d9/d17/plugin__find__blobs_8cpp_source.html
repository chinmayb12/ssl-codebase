46  CMVision::RegionList * reglist;

55 
56  CMVision::RunList * runlist;

58  printf("Blob finder: no runlength-encoded input list was found!\n");

60  }
61 
62  if (_v_enable->getBool()==true) {
63  //Connect the components of the runlength map:
64  CMVision::RegionProcessing::connectComponents(runlist);

67  CMVision::RegionProcessing::extractRegions(reglist, runlist);
68 
69  if (reglist->getUsedRegions() == reglist->getMaxRegions()) {
70  printf("Warning: extract regions exceeded maximum number of %d regions\n",reglist->getMaxRegions());
71  }
72

75 
76  //Sort Regions:

78  } else {

87  }
