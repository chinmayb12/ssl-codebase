137 double NKD_FUN::distFromQuery(Node *p)
138 {
139  vec_t near;
140  near.bound(query_point, p->min, p->max);
141  if(!scaled){
142  return(dist(query_point, near));
143  }else{
144  double d = 0.0;
145  for(int i=0; i<dim; i++){
146  d += sq((query_point[i] - near[i]) * scale[i]);
147  }
148  return(sqrt(d));
149  }
150 }

227 void NKD_FUN::add(Node **q,state_t *s,int level)
228 {
229  get_next next;
230  Node *p = *q;
231 
232  // check if p is a leaf node
233  if(!p || !p->child[0]){
234  // we're at a leaf, so add state to this node
235 
236  // make a new node if none exists here
237  if(!p){
238  *q = p = new Node;
239  mzero(*p);
240  initBBox(*p,*s);
241  }
242 
243  // add point
244  next(s) = p->states;
245  p->states = s;
246  p->num_states++;
247  updateBBox(*p,*s);
248 
249  // if tree has already been built, split leaf if necessary
250  if(is_built) split(p,level);
251  }else{
252  // there is an existing tree and we're at an internal node, so
253  // just update out bounding box and recurse through this node
254  updateBBox(*p,*s);
255  int i = (*s)[p->split_dim] > p->threshold;
256  add(&p->child[i],s,level+1);
257  }

260 NKD_TEM
261 void NKD_FUN::split(Node *p,int level)
262 {
263  get_next next;
264  state_t *s,*sn;
265  int sd;
266  double sum;
267  int i;
268 
269  // check if we actually want to split
270  if(!p) return;
271  if((p->num_states <= leaf_size) || (level >= max_depth)) return;
272  if(sqdist(p->min,p->max) < 1E-9) return;
273 
274  // choose split dimension as round-robin
275  p->split_dim = sd = level % dim;
276 
277  // calculate mean for split threshold
278  sum = 0.0;
279  s = p->states;
280  while(s){
281  sum += (*s)[sd];
282  s = next(s);
283  }
284  p->threshold = sum / p->num_states;
285 
286  // split the points based on the plane
287  s = p->states;
288  while(s){
289  sn = next(s);
290 
291  i = (*s)[sd] > p->threshold;
292  add(&(p->child[i]),s,level+1);
293 
294  s = sn;
295  }
296 
297  // this state no longer stores points
298  p->states = 0;
299  p->num_states = 0;
300 
301  if(!is_built){
302  split(p->child[0],level+1);
303  split(p->child[1],level+1);
304  }
305 }

359 state_t *NKD_FUN::getNextNearest(double &dist)
360 {
361  get_next next;
362  QueryNode qn;
363  state_t *s;
364 
365  while(queue.size() > 0){
366  // get head of priority queue
367  qn = queue.top();
368  queue.pop();
369 
370  // if its a raw state, return it
371  if(qn.state){
372  dist = qn.dist;
373  return(qn.state);
374  }
375 
376  // otherwise it is a node, so we have to expand it
377 
378  // expand nodes
379  if(qn.node->child[0]) addToSearchQueue(qn.node->child[0]);
380  if(qn.node->child[1]) addToSearchQueue(qn.node->child[1]);
381 
382  // expand states
383  s = qn.node->states;
384  while(s){
385  addToSearchQueue(s);
386  s = next(s);
387  }
388  }
389 
390  // ran out of states
391  return(NULL);
392 }

395 state_t *NKD_FUN::getOnlyNearest(Node *t,state_t *nearest)
396 {
397  get_next next;
398  state_t *s;
399  double d;
400 
401  // pruning
402  if(t==NULL || distFromQuery(t)>query_max_dist) return(nearest);
403 
404  // we either store states (leaf) or are have children (interal node)
405  if(t->states){
406  // leaf node, check the states stored here to update the nearest
407  s = t->states;
408  while(s){
409  d = distFromQuery(s);
410  if(d < query_max_dist){
411  nearest = s;
412  query_max_dist = d;
413  }
414  s = next(s);
415  }
416  }else{
417  // internal node, find out which side of the plane our query is on
418  // and descend the tree on that side first, followed by the
419  // further side.
420  int i = query_point[t->split_dim] > t->threshold;
421  nearest = getOnlyNearest(t->child[ i],nearest);
422  nearest = getOnlyNearest(t->child[!i],nearest);
423  }
424 
425  return(nearest);
426 }
