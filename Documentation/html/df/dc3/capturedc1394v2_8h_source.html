67  #endif
68  protected:
69  char counter;
70  dc1394_t* dc_instance;
71  public:
72  dc1394_t* obtainInstance() {
73  lock();
74  if (counter == 0) {
75  dc_instance=dc1394_new();
76  }
77  counter++;
78  dc1394_t* inst=dc_instance;
79 
80  unlock();
81  return inst;
82  }
83  bool removeInstance() {
84  lock();
85  counter--;
86  bool was_last=false;
87  if (counter == 0) {
88  dc1394_free( dc_instance );
89  was_last=true;
90  }
91  if (counter < 0) {
92  fprintf(stderr,"WARNING: Attempting to remove more instances than have been created!\n");
93  fflush(stderr);
94  counter = 0;
95  }
96  unlock();
97  return was_last;
98  }
99 
100  private:
101  void lock() {
102  #ifndef VDATA_NO_QT
103  mutex.lock();
104  #else
105  pthread_mutex_lock();
106  #endif
107  }
108  void unlock() {
109  #ifndef VDATA_NO_QT
110  mutex.unlock();
111  #else
112  pthread_mutex_unlock();
113  #endif
114  }
