151  bool update(GLint wrap_s=GL_CLAMP, GLint wrap_t=GL_CLAMP, GLint minfilter=GL_LINEAR, GLint magfilter=GL_LINEAR) {
152  if (loaded==false) return false;

154  glBindTexture( texTarget, texture );
155  //printf("glupdatebind err: %s\n",gluErrorString(glGetError()));

159  glTexImage2D( texTarget, 0, 3, surface.getWidth(),
160  surface.getHeight(), 0, GL_RGB,

162 
163  } else if (surface.getColorFormat()==COLOR_RGBA8) {
164  // generate texture from bitmap

166  surface.getHeight(), 0, GL_RGBA,

169  //printf("glteximage2d rgba err: %s\n",gluErrorString(glGetError()));
170  } else {
171  fprintf(stderr,"Error: unknown image format to convert to OpenGL\n");
172  }
173  // Linear Filtering
174  glTexParameteri( texTarget, GL_TEXTURE_MIN_FILTER, minfilter );
175  glTexParameteri( texTarget, GL_TEXTURE_MAG_FILTER, magfilter );
176  glTexParameteri( texTarget, GL_TEXTURE_WRAP_S, wrap_s);
177  glTexParameteri( texTarget, GL_TEXTURE_WRAP_T, wrap_t);
178  return true;

201  {

204  //reload texture to RAM

210

213

215  //now sync data to video card
