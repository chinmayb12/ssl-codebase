73 bool PluginDetectBalls::checkHistogram ( const Image<raw8> * image, const CMVision::Region * reg, double min_greenness, double max_markeryness ) {
74  static const int PixelRadius = 4;
75 
76  histogram->clear();
77 
78  int num = histogram->addBox ( image, reg->x1 - PixelRadius, reg->y1 - PixelRadius,
79  reg->x2 + PixelRadius, reg->y2 + PixelRadius );
80 
81 
82  float pf = ( float ) ( histogram->getChannel ( color_id_pink ) ) / ( float ) ( histogram->getChannel ( color_id_orange ) );
83  float yf = ( float ) ( histogram->getChannel ( color_id_yellow ) ) / ( float ) ( histogram->getChannel ( color_id_orange ) );
84  float markeryness = ( pf + 1 ) * ( yf + 1 ) - 1;
85  float greenness = ( float ) ( histogram->getChannel ( color_id_field ) ) / ( ( ( float ) ( num - histogram->getChannel ( color_id_orange ) ) ) + 1E-6 );
86 
87  if ( greenness > min_greenness ) return ( true );
88  if ( markeryness > max_markeryness ) return ( false );
89  return ( true );
90 }

109 ProcessResult PluginDetectBalls::process ( FrameData * data, RenderOptions * options ) {
110  ( void ) options;
111  if ( data==0 ) return ProcessingFailed;
112 
113  SSL_DetectionFrame * detection_frame = 0;
114 
115  detection_frame= ( SSL_DetectionFrame * ) data->map.get ( "ssl_detection_frame" );
116  if ( detection_frame == 0 ) detection_frame= ( SSL_DetectionFrame * ) data->map.insert ( "ssl_detection_frame",new SSL_DetectionFrame() );
117 
118  int color_id_ball = _lut->getChannelID ( _settings->_color_label->getString() );
119  if ( color_id_ball == -1 ) {
120  printf ( "Unknown Ball Detection Color Label: '%s'\nAborting Plugin!\n",_settings->_color_label->getString().c_str() );
121  return ProcessingFailed;
122  }
123 
124  //delete any previous detection results:
125  detection_frame->clear_balls();
126 
127  //TODO: add a vartype notifier for better performance.
128 
129  //initialize filter:
130  if (vnotify.hasChanged()) {
131  max_balls = _settings->_max_balls->getInt();
132  filter.setWidth ( _settings->_ball_min_width->getInt(),_settings->_ball_max_width->getInt() );
133  filter.setHeight ( _settings->_ball_min_height->getInt(),_settings->_ball_max_height->getInt() );
134  filter.setArea ( _settings->_ball_min_area->getInt(),_settings->_ball_max_area->getInt() );
135  field_filter.update ( field );
136 
137  //copy all vartypes to local variables for faster repeated lookup:
138  filter_ball_in_field = _settings->_ball_on_field_filter->getBool();
139  filter_ball_on_field_filter_threshold = _settings->_ball_on_field_filter_threshold->getDouble();
140  filter_ball_in_goal = _settings->_ball_in_goal_filter->getBool();
141  filter_ball_histogram = _settings->_ball_histogram_enabled->getBool();
142  if ( filter_ball_histogram ) {
143  if ( color_id_ball != color_id_orange ) {
144  printf ( "Warning: ball histogram check is only configured for orange balls!\n" );
145  printf ( "Please disable the histogram check in the Ball Detection Plugin settings\n" );
146  }
147  if ( color_id_pink==-1 || color_id_orange==-1 || color_id_yellow==-1 || color_id_field==-1 ) {
148  printf ( "WARNING: some LUT color labels where undefined for the ball detection plugin\n" );
149  printf ( " Disabling histogram check!\n" );
150  filter_ball_histogram=false;
151  }
152  }
153 
154  min_greenness = _settings->_ball_histogram_min_greenness->getDouble();
155  max_markeryness = _settings->_ball_histogram_max_markeryness->getDouble();
156 
157  //setup values used for the gaussian confidence measurement:
158  filter_gauss = _settings->_ball_gauss_enabled->getBool();
159  exp_area_min = _settings->_ball_gauss_min->getInt();
160  exp_area_max = _settings->_ball_gauss_max->getInt();
161  exp_area_var = sq ( _settings->_ball_gauss_stddev->getDouble() );
162  z_height= _settings->_ball_z_height->getDouble();
163 
164  near_robot_filter = _settings->_ball_too_near_robot_enabled->getBool();
165  near_robot_dist_sq = sq(_settings->_ball_too_near_robot_dist->getDouble());
166  }
167 
168  const CMVision::Region * reg = 0;
169 
170  //acquire orange region list from data-map:
171  CMVision::ColorRegionList * colorlist;
172  colorlist= ( CMVision::ColorRegionList * ) data->map.get ( "cmv_colorlist" );
173  if ( colorlist==0 ) {
174  printf ( "error in ball detection plugin: no region-lists were found!\n" );
175  return ProcessingFailed;
176  }
177  reg = colorlist->getRegionList ( color_id_ball ).getInitialElement();
178 
179  //acquire color-labeled image from data-map:
180  const Image<raw8> * image = ( Image<raw8> * ) ( data->map.get ( "cmv_threshold" ) );
181  if ( image==0 ) {
182  printf ( "error in ball detection plugin: no color-thresholded image was found!\n" );
183  return ProcessingFailed;
184  }
185 
186  int robots_blue_n=0;
187  int robots_yellow_n=0;
188  bool use_near_robot_filter=near_robot_filter;
189  if ( use_near_robot_filter ) {
190  SSL_DetectionFrame * detection_frame = ( SSL_DetectionFrame * ) data->map.get ( "ssl_detection_frame" );
191  if ( detection_frame==0 ) {
192  use_near_robot_filter=false;
193  } else {
194  robots_blue_n=detection_frame->robots_blue_size();
195  robots_yellow_n=detection_frame->robots_yellow_size();
196  if (robots_blue_n==0 && robots_yellow_n==0) use_near_robot_filter=false;
197  }
198  }
199 
200  if ( max_balls > 0 ) {
201  list<BallDetectResult> result;
202  filter.init ( reg );
203 
204  while ( ( reg = filter.getNext() ) != 0 ) {
205  float conf = 1.0;
206 
207  if ( filter_gauss==true ) {
208  int a = reg->area - bound ( reg->area,exp_area_min,exp_area_max );
209  conf = gaussian ( a / exp_area_var );
210  }
211 
212  //TODO: add a plugin for confidence masking... possibly multi-layered.
213  // to replace the commented det.mask.get(...) below:
214  //if (filter_conf_mask) conf*=det.mask.get(reg->cen_x,reg->cen_y));
215 
216  //convert from image to field coordinates:
217  vector2d pixel_pos ( reg->cen_x,reg->cen_y );
218  vector3d field_pos_3d;
219  camera_parameters.image2field ( field_pos_3d,pixel_pos,z_height );
220  vector2d field_pos ( field_pos_3d.x,field_pos_3d.y );
221 
222  //filter points that are outside of the field:
223  if ( filter_ball_in_field==true && field_filter.isInFieldPlusThreshold ( field_pos, max(0.0,filter_ball_on_field_filter_threshold) ) ==false ) {
224  conf = 0.0;
225  }
226 
227  //filter out points that are deep inside the goal-box
228  if ( filter_ball_in_goal==true && field_filter.isFarInGoal ( field_pos ) ==true ) {
229  conf = 0.0;
230  }
231 
232  //TODO add ball-too-near-robot filter
233  if ( use_near_robot_filter && conf > 0.0 ) {
234  int robots_n=0;
235  for (int team = 0; team < 2; team++) {
236  if (team==0) {
237  robots_n=robots_blue_n;
238  } else {
239  robots_n=robots_yellow_n;
240  }
241  if (robots_n > 0) {
242  ::google::protobuf::RepeatedPtrField< ::SSL_DetectionRobot > * robots;
243  if (team==0) {
244  robots = detection_frame->mutable_robots_blue();
245  } else {
246  robots = detection_frame->mutable_robots_yellow();
247  }
248  for (int r = 0; r < robots_n; r++) {
249  const SSL_DetectionRobot & robot = robots->Get(r);
250  if (robot.confidence() > 0.0) {
251  if ((sq((double)(robot.x())-(double)(field_pos.x)) + sq((double)(robot.y())-(double)(field_pos.y))) < near_robot_dist_sq) {
252  conf = 0.0;
253  break;
254  }
255  }
256  }
257  if (conf==0.0) break;
258  }
259  }
260  }
261 
262  // histogram check if enabled
263  if ( filter_ball_histogram && conf > 0.0 && checkHistogram ( image, reg, min_greenness, max_markeryness ) ==false ) {
264  conf = 0.0;
265  }
266 
267  // add filtered region to the region list
268  if(conf > 0) {
269  result.push_back(BallDetectResult(reg,conf));
270  }
271 
272  }
273 
274  // sort result by confidence and output first max_balls region(s)
275  result.sort();
276 
277  int num_ball = 0;
278  list<BallDetectResult>::reverse_iterator it;
279  for(it=result.rbegin(); it!=result.rend(); it++) {
280  if(++num_ball > max_balls)
281  break;
282 
283  //update result:
284  SSL_DetectionBall* ball = detection_frame->add_balls();
285 
286  ball->set_confidence ( it->conf );
287 
288  vector2d pixel_pos ( it->reg->cen_x,it->reg->cen_y );
289  vector3d field_pos_3d;
290  camera_parameters.image2field ( field_pos_3d,pixel_pos,z_height );
291 
292  ball->set_area ( it->reg->area );
293  ball->set_x ( field_pos_3d.x );
294  ball->set_y ( field_pos_3d.y );
295  ball->set_pixel_x ( it->reg->cen_x );
296  ball->set_pixel_y ( it->reg->cen_y );
297  }
298 
299  }
300 
301  return ProcessingOk;
302 
303 }
