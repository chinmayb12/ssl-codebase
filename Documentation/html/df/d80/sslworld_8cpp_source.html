916 /******************************************************************************/
917 /* randn()
918  *
919  * Normally (Gaussian) distributed random numbers, using the Box-Muller
920  * transformation. This transformation takes two uniformly distributed deviates
921  * within the unit circle, and transforms them into two independently
922  * distributed normal deviates. Utilizes the internal rand() function; this can
923  * easily be changed to use a better and faster RNG.
924  *
925  * The parameters passed to the function are the mean and standard deviation of
926  * the desired distribution. The default values used, when no arguments are
927  * passed, are 0 and 1 - the standard normal distribution.
928  *
929  *
930  * Two functions are provided:
931  *
932  * The first uses the so-called polar version of the B-M transformation, using
933  * multiple calls to a uniform RNG to ensure the initial deviates are within the
934  * unit circle. This avoids making any costly trigonometric function calls.
935  *
936  * The second makes only a single set of calls to the RNG, and calculates a
937  * position within the unit circle with two trigonometric function calls.
938  *
939  * The polar version is generally superior in terms of speed; however, on some
940  * systems, the optimization of the math libraries may result in better
941  * performance of the second. Try it out on the target system to see which
942  * works best for you. On my test machine (Athlon 3800+), the non-trig version
943  * runs at about 3x10^6 calls/s; while the trig version runs at about
944  * 1.8x10^6 calls/s (-O2 optimization).
945  *
946  *
947  * Example calls:
948  * randn_notrig(); //returns normal deviate with mean=0.0, std. deviation=1.0
949  * randn_notrig(5.2,3.0); //returns deviate with mean=5.2, std. deviation=3.0
950  *
951  *
952  * Dependencies - requires <cmath> for the sqrt(), sin(), and cos() calls, and a
953  * #defined value for PI.
954  */
955 
956 /******************************************************************************/
957 // "Polar" version without trigonometric calls
958 dReal randn_notrig(dReal mu, dReal sigma) {
959  if (sigma==0) return mu;
960  static bool deviateAvailable=false; // flag
961  static dReal storedDeviate; // deviate from previous calculation
962  dReal polar, rsquared, var1, var2;
963 
964  // If no deviate has been stored, the polar Box-Muller transformation is
965  // performed, producing two independent normally-distributed random
966  // deviates. One is stored for the next round, and one is returned.
967  if (!deviateAvailable) {
968 
969  // choose pairs of uniformly distributed deviates, discarding those
970  // that don't fall within the unit circle
971  do {
972  var1=2.0*( dReal(rand())/dReal(RAND_MAX) ) - 1.0;
973  var2=2.0*( dReal(rand())/dReal(RAND_MAX) ) - 1.0;
974  rsquared=var1*var1+var2*var2;
975  } while ( rsquared>=1.0 || rsquared == 0.0);
976 
977  // calculate polar tranformation for each deviate
978  polar=sqrt(-2.0*log(rsquared)/rsquared);
979 
980  // store first deviate and set flag
981  storedDeviate=var1*polar;
982  deviateAvailable=true;
983 
984  // return second deviate
985  return var2*polar*sigma + mu;
986  }
987 
988  // If a deviate is available from a previous call to this function, it is
989  // returned, and the flag is set to false.
990  else {
991  deviateAvailable=false;
992  return storedDeviate*sigma + mu;
993  }
994 }
