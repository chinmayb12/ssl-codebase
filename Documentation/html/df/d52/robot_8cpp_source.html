132 bool Robot::Kicker::isTouchingBall()
133 {
134  dReal vx,vy,vz;
135  dReal bx,by,bz;
136  dReal kx,ky,kz;
137  rob->chassis->getBodyDirection(vx,vy,vz);
138  rob->getBall()->getBodyPosition(bx,by,bz);
139  box->getBodyPosition(kx,ky,kz);
140  kx += vx*rob->cfg->robotSettings.KickerThickness*0.5f;
141  ky += vy*rob->cfg->robotSettings.KickerThickness*0.5f;
142  dReal xx = fabs((kx-bx)*vx + (ky-by)*vy);
143  dReal yy = fabs(-(kx-bx)*vy + (ky-by)*vx);

165 void Robot::Kicker::kick(dReal kickspeedx, dReal kickspeedz)
166 {
167  dReal dx,dy,dz;
168  dReal vx,vy,vz;
169  rob->chassis->getBodyDirection(dx,dy,dz);dz = 0;
170  dReal zf = kickspeedz;
171  if (isTouchingBall())
172  {
173  dReal dlen = dx*dx+dy*dy+dz*dz;
174  dlen = sqrt(dlen);
175  vx = dx*kickspeedx/dlen;
176  vy = dy*kickspeedx/dlen;
177  vz = zf;
178  const dReal* vball = dBodyGetLinearVel(rob->getBall()->body);
179  dReal vn = -(vball[0]*dx + vball[1]*dy)*rob->cfg->robotSettings.KickerDampFactor;
180  dReal vt = -(vball[0]*dy - vball[1]*dx);
181  vx += vn * dx - vt * dy;
182  vy += vn * dy + vt * dx;
183  dBodySetLinearVel(rob->getBall()->body,vx,vy,vz);
184  }
185  kicking = true;
186  kickstate = 10;
187 }
