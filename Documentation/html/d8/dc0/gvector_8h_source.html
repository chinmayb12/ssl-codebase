354 // returns point rotated around axis <r> by <a> radians (right handed)
355 template <class num>
356 vector3d<num> vector3d<num>::rotate(const vector3d<num> r,const double a) const
357 {
358  vector3d<num> q;
359  double s,c,t;
360 
361  s = sin(a);
362  c = cos(a);
363  t = 1 - c;
364 
365  q.x = (t * r.x * r.x + c ) * x
366  + (t * r.x * r.y - s * r.z) * y
367  + (t * r.x * r.z + s * r.y) * z;
368 
369  q.y = (t * r.y * r.x + s * r.z) * x
370  + (t * r.y * r.y + c ) * y
371  + (t * r.y * r.z - s * r.x) * z;
372 
373  q.z = (t * r.z * r.x - s * r.y) * x
374  + (t * r.z * r.y + s * r.x) * y
375  + (t * r.z * r.z + c ) * z;
376 
377  return(q);
378 }
