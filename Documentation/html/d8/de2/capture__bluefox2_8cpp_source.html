193  if(hdrMode == HDR_MODE_OFF)

325 bool CaptureBlueFox2::startCapture()
326 {
327  #ifndef VDATA_NO_QT
328  mutex.lock();
329  #endif
330 
331  //grab current parameters:
332  cam_id = v_cam_bus->getInt();
333 
334  const unsigned int devCnt = devMgr.deviceCount();
335  fprintf(stderr, "BlueFox2: Number of cams: %u\n", devCnt);
336 
337  if(cam_id >= devCnt)
338  {
339  fprintf(stderr, "BlueFox2: Invalid cam_id: %u\n", cam_id);
340 
341  #ifndef VDATA_NO_QT
342  mutex.unlock();
343  #endif
344  return false;
345  }
346 
347  pDevice = devMgr[cam_id];
348 
349  try
350  {
351  pDevice->open();
352  }
353  catch(mvIMPACT::acquire::ImpactAcquireException& e)
354  {
355  // this e.g. might happen if the same device is already opened in another process...
356  fprintf(stderr, "BlueFox2: An error occurred while opening the device(error code: %d, '%s')\n", e.getErrorCode(), e.getErrorString().c_str());
357  #ifndef VDATA_NO_QT
358  mutex.unlock();
359  #endif
360  return false;
361  }
362 
363  fprintf(stderr, "BlueFox2: Opened: %s with serial ID %s\n", pDevice->family.read().c_str(), pDevice->serial.read().c_str());
364 
365  pFI = new FunctionInterface(pDevice);
366 
367  ImageDestination id( pDevice );
368  id.restoreDefault();
369 
370  ColorFormat out_color = Colors::stringToColorFormat(v_colorout->getSelection().c_str());
371  if(out_color == COLOR_RGB8)
372  {
373  id.pixelFormat.write(idpfBGR888Packed);
374  }
375  else
376  {
377  id.pixelFormat.write(idpfYUV422Packed);
378  }
379 
380  pSettings = new CameraSettingsBlueFOX(pDevice);
381  pSettings->restoreDefault();
382 
383  pImageProc = new ImageProcessing(pDevice);
384  pImageProc->restoreDefault();
385 
386  is_capturing = true;
387 
388  vector<VarType *> tmp = capture_settings->getChildren();
389  for (unsigned int i=0; i < tmp.size();i++) {
390  tmp[i]->addFlags( VARTYPE_FLAG_READONLY );
391  }
392 
393  dcam_parameters->removeFlags( VARTYPE_FLAG_HIDE_CHILDREN );
394 
395  #ifndef VDATA_NO_QT
396  mutex.unlock();
397  #endif
398 
399  printf("BlueFox2 Info: Restoring Previously Saved Camera Parameters\n");
400  writeAllParameterValues();
401  readAllParameterValues();
402 
403  return true;
404 }

406 bool CaptureBlueFox2::copyAndConvertFrame(const RawImage & src, RawImage & target)
407 {
408  #ifndef VDATA_NO_QT
409  mutex.lock();
410  #endif
411 
412  ColorFormat src_fmt = src.getColorFormat();
413 
414  if(target.getData() == 0)
415  {
416  //allocate target, if it does not exist yet
417  target.allocate(src_fmt, src.getWidth(), src.getHeight());
418  }
419  else
420  {
421  target.ensure_allocation(src_fmt, src.getWidth(), src.getHeight());
422  }
423  target.setTime(src.getTime());
424 
425  if(src.getColorFormat() == COLOR_RGB8)
426  {
427  memcpy(target.getData(),src.getData(),src.getNumBytes());
428  }
429  else
430  {
431  for(int i = 0; i < src.getNumBytes(); i += 2)
432  {
433  target.getData()[i+1] = src.getData()[i];
434  target.getData()[i] = src.getData()[i+1];
435  }
436  }
437 
438  #ifndef VDATA_NO_QT
439  mutex.unlock();
440  #endif
441 
442  return true;
443 }

445 RawImage CaptureBlueFox2::getFrame()
446 {
447  #ifndef VDATA_NO_QT
448  mutex.lock();
449  #endif
450 
451  RawImage result;
452  result.setColorFormat(capture_format);
453  result.setWidth(0);
454  result.setHeight(0);
455  result.setTime(0.0);
456  result.setData(0);
457 
458  // make sure the request queue is always filled
459  while((static_cast<TDMR_ERROR>( pFI->imageRequestSingle() ) ) == DMR_NO_ERROR ) {};
460 
461  int requestNr = pFI->imageRequestWaitFor(-1);
462 
463  // check if the image has been captured without any problems.
464  if(!pFI->isRequestNrValid(requestNr))
465  {
466  // If the error code is -2119(DEV_WAIT_FOR_REQUEST_FAILED), the documentation will provide
467  // additional information under TDMR_ERROR in the interface reference
468  fprintf(stderr, "imageRequestWaitFor failed (%d, %s)\n", requestNr, ImpactAcquireException::getErrorCodeAsString( requestNr ).c_str());
469  #ifndef VDATA_NO_QT
470  mutex.unlock();
471  #endif
472  return result;
473  }
474 
475  const Request* pRequest = pFI->getRequest(requestNr);
476 
477  if(pRequest->isOK())
478  {
479  timeval tv;
480  gettimeofday(&tv,NULL);
481  result.setTime((double)tv.tv_sec + tv.tv_usec*(1.0E-6));
482  result.setWidth(pRequest->imageWidth.read());
483  result.setHeight(pRequest->imageHeight.read());
484  ColorFormat out_color = Colors::stringToColorFormat(v_colorout->getSelection().c_str());
485  result.setColorFormat(out_color);
486  result.setData((unsigned char*)pRequest->imageData.read());
487 // fprintf(stderr, "Timestamp_us: %ld\n", pRequest->infoTimeStamp_us.read());
488 // fprintf(stderr, "BPP: %u\n", pRequest->imageBytesPerPixel.read());
489 // fprintf(stderr, "PixelFormat: %s\n", pRequest->imagePixelFormat.readS().c_str());
490  }
491  else
492  {
493  fprintf(stderr, "BlueFox2: request not OK\n");
494  }
495 
496  lastRequestNr = requestNr;
497 
498  #ifndef VDATA_NO_QT
499  mutex.unlock();
500  #endif
501  return result;
502 }
