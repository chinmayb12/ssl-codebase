83 void PluginLegacyPublishGeometry::sendGeometry() {
84  // NOTE: The field dimensions are derived from one half irrespective of
85  // the exact field half / end in question.
86  SSL_GeometryFieldSize field;
87  _field.toProtoBuffer(field);
88 
89  // Geometry data for double-sized field in old format.
90  RoboCup2014Legacy::Geometry::SSL_GeometryData ds_geodata_old;
91  // Field data for double-sized field in old format.
92  RoboCup2014Legacy::Geometry::SSL_GeometryFieldSize ds_field_old;
93 
94  ds_field_old.set_field_length(field.field_length());
95  ds_field_old.set_field_width(field.field_width());
96  ds_field_old.set_goal_width(field.goal_width());
97  ds_field_old.set_goal_depth(field.goal_depth());
98  ds_field_old.set_boundary_width(field.boundary_width());
99  ds_field_old.set_center_circle_radius(
100  GetFieldCircularArcRadius("CenterCircle"));
101  ds_field_old.set_defense_radius(
102  GetFieldCircularArcRadius("LeftFieldLeftPenaltyArc"));
103  ds_field_old.set_defense_stretch(GetFieldLineLength("LeftPenaltyStretch"));
104  // The following fields no longer exist in ssl-vision, and are hard-coded
105  // from the ssl-vision SVN trunk, r233.
106  ds_field_old.set_line_width(10);
107  ds_field_old.set_referee_width(425);
108  ds_field_old.set_goal_wall_width(20);
109  ds_field_old.set_free_kick_from_defense_dist(200);
110  ds_field_old.set_penalty_spot_from_field_line_dist(750);
111  ds_field_old.set_penalty_line_from_spot_dist(400);
112  *ds_geodata_old.mutable_field() = ds_field_old;
113 
114 
115  // Copy over the camera calibrations.
116  for (unsigned int i = 0; i < params.size(); i++) {
117  SSL_GeometryCameraCalibration* ds_calib_old = ds_geodata_old.add_calib();
118  params[i]->toProtoBuffer(*ds_calib_old,i);
119  }
120 
121  _ds_udp_server_old->sendLegacyMessage(ds_geodata_old);
122 }

124 float PluginLegacyPublishGeometry::GetFieldLineLength(const string& line_name) {
125  const FieldLine* line = GetFieldLine(line_name, _field.field_lines);
126  if (line == NULL) return 0;
